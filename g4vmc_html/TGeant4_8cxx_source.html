51 Bool_t CheckApplicationState(const TString& methodName,
52  TG4ApplicationState requiredState, Bool_t allowLater = false,
53  Bool_t allowSooner = false, Bool_t allowJustAfter = false)
54 {
58 
59  TG4StateManager* stateManager = TG4StateManager::Instance();
60  TG4ApplicationState currentState = stateManager->GetCurrentState();
61  TG4ApplicationState previousState = stateManager->GetPreviousState();
62  if (currentState != requiredState &&
63  !((allowLater && currentState > requiredState) ||
64  (allowLater && previousState >= requiredState)) &&
65  !((allowSooner && currentState < requiredState) ||
66  (allowSooner && previousState <= requiredState)) &&
67  !(allowJustAfter && previousState == requiredState)) {
68 
69  TString message =
70  TString("MC::") + methodName +
71  +TString(" can be called only from VMCApplication::") +
72  TString(stateManager->GetStateName(requiredState).c_str());
73  if (allowLater) message += TString(" or after");
74  if (allowSooner) message += TString(" or before");
75  if (allowJustAfter) message += TString(" or just after");
76  message += TG4Globals::Endl() +
77  TString("while detected in VMCApplication::") +
78  TString(stateManager->GetStateName(currentState).c_str());
79  if (currentState == kNotInApplication)
80  message += TString(" after VMCApplication::") +
81  TString(stateManager->GetStateName(previousState).c_str());
82 
83  TG4Globals::Warning("TGeant4", methodName, message);
84  return false;
85  }
86 
87  return true;
88 }

91 Bool_t CheckG4ApplicationState(const TString& methodName,
92  G4ApplicationState requiredState, Bool_t allowLater = false)
93 {
97 
98  G4ApplicationState currentState =
99  G4StateManager::GetStateManager()->GetCurrentState();
100 
101  if (currentState != requiredState &&
102  !(allowLater && currentState > requiredState)) {
103 
104  // States as strings
105  static std::vector<TString> g4StateNames;
106  g4StateNames.push_back("G4State_PreInit");
107  g4StateNames.push_back("G4State_Init");
108  g4StateNames.push_back("G4State_Idle");
109  g4StateNames.push_back("G4State_GeomClosed");
110  g4StateNames.push_back("G4State_EventProc");
111  g4StateNames.push_back("G4State_Quit");
112  g4StateNames.push_back("G4State_Abort");
113 
114  TString message = TString("MC::") + methodName +
115  +TString(" can be called only when Geant4 is in state ") +
116  g4StateNames[requiredState];
117  if (allowLater) message += TString(" or later");
118  message += TG4Globals::Endl() + TString("while detected in state ") +
119  g4StateNames[currentState];
120 
121  TG4Globals::Warning("TGeant4", methodName, message);
122  return false;
123  }
124 
125  return true;
126 }

199  // add verbose level

215  // add verbose level
