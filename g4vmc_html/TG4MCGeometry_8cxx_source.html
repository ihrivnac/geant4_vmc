163  // !!! in Geant3:

668 Bool_t TG4MCGeometry::GetTransformation(
669  const TString& volumePath, TGeoHMatrix& matrix)
670 {
673 
674  G4String path = volumePath.Data();
675 
676  G4String volName;
677  G4int copyNo;
678  path = fGeometryServices->CutVolumePath(path, volName, copyNo);
679 
680  // Get the first volume
681  G4VPhysicalVolume* pvTop = 0;
682  if (fGeometryServices->GetWorld()->GetName() == volName) {
683  // Check world volume first
684  pvTop = fGeometryServices->GetWorld();
685  }
686  else
687  pvTop = fGeometryServices->FindPhysicalVolume(volName, copyNo, true);
688 
689  if (!pvTop) {
690  TG4Globals::Warning("TG4MCGeometry",
691  "GetTransformation:", "Top volume in " + volumePath + " does not exist.");
692  return false;
693  }
694 
695  // Calculate transformation
696  //
697  G4VPhysicalVolume* pvMother = pvTop;
698  G4Transform3D transform;
699 
700  while (path.length() > 0) {
701  // Extract next volume name & copyNo
702  path = fGeometryServices->CutVolumePath(path, volName, copyNo);
703 
704  // Find daughter
705  G4VPhysicalVolume* pvDaughter = fGeometryServices->FindDaughter(
706  volName, copyNo, pvMother->GetLogicalVolume(), true);
707 
708  if (!pvDaughter) {
709  TG4Globals::Warning("TG4MCGeometry", "GetTransformation",
710  "Daughter volume " + TString(volName) + " in " + volumePath +
711  " does not exist.");
712  return false;
713  }
714 
715  transform = transform * G4Transform3D(*pvDaughter->GetObjectRotation(),
716  pvDaughter->GetObjectTranslation());
717  pvMother = pvDaughter;
718  }
719 
720  // put transform in TGeoHMatrix here
721  fGeometryServices->Convert(transform, matrix);
722  return true;
723 }

726 Bool_t TG4MCGeometry::GetShape(
727  const TString& volumePath, TString& shapeType, TArrayD& par)
728 {
731 
732  // Get volume & copyNo
733  G4String path = volumePath.Data();
734  G4int last1 = path.rfind('/');
735  G4int last2 = path.rfind('_');
736  G4String volName = path(last1 + 1, last2 - last1 - 1);
737  G4String copyNoStr = path(last2 + 1, path.length() - last2);
738  std::istringstream in(copyNoStr);
739  G4int copyNo;
740  in >> copyNo;
741 
742  // Find physical volume
743  G4VPhysicalVolume* pv =
744  fGeometryServices->FindPhysicalVolume(volName, copyNo, true);
745 
746  if (!pv) {
747  TG4Globals::Warning("TG4MCGeometry", "GetShape",
748  "Physical volume " + volumePath + " does not exist.");
749  return false;
750  }
751 
752  // Get solid
753  G4VSolid* solid = pv->GetLogicalVolume()->GetSolid();
754 
755  // Get constituent solid if reflected solid
756  Bool_t isReflected = false;
757  G4ReflectedSolid* reflSolid = dynamic_cast<G4ReflectedSolid*>(solid);
758  if (reflSolid) {
759  solid = reflSolid->GetConstituentMovedSolid();
760  isReflected = true;
761  }
762 
763  Int_t npar;
764 
765  if (solid->GetEntityType() == "G4Box") {
766  shapeType = "BOX";
767  npar = 3;
768  par.Set(npar);
769  G4Box* box = (G4Box*)solid;
770  par.AddAt(box->GetXHalfLength() / cm, 0);
771  par.AddAt(box->GetYHalfLength() / cm, 1);
772  par.AddAt(box->GetZHalfLength() / cm, 2);
773  return kTRUE;
774  }
775 
776  if (solid->GetEntityType() == "G4Cons") {
777  shapeType = "CONS";
778  npar = 7;
779  par.Set(npar);
780  G4Cons* cons = (G4Cons*)solid;
781  par.AddAt(cons->GetZHalfLength() / cm, 0);
782  if (!isReflected) {
783  par.AddAt(cons->GetInnerRadiusMinusZ() / cm, 1);
784  par.AddAt(cons->GetOuterRadiusMinusZ() / cm, 2);
785  par.AddAt(cons->GetInnerRadiusPlusZ() / cm, 3);
786  par.AddAt(cons->GetOuterRadiusPlusZ() / cm, 4);
787  }
788  else {
789  par.AddAt(cons->GetInnerRadiusPlusZ() / cm, 1);
790  par.AddAt(cons->GetOuterRadiusPlusZ() / cm, 2);
791  par.AddAt(cons->GetInnerRadiusMinusZ() / cm, 3);
792  par.AddAt(cons->GetOuterRadiusMinusZ() / cm, 4);
793  }
794  par.AddAt(cons->GetStartPhiAngle() / deg, 5);
795  par.AddAt(cons->GetDeltaPhiAngle() / deg, 6);
796  return kTRUE;
797  }
798 
799  if (solid->GetEntityType() == "G4EllipticalTube") {
800  shapeType = "ELTU";
801  npar = 3;
802  par.Set(npar);
803  G4EllipticalTube* eltu = (G4EllipticalTube*)solid;
804  par.AddAt(eltu->GetDx() / cm, 0);
805  par.AddAt(eltu->GetDy() / cm, 1);
806  par.AddAt(eltu->GetDz() / cm, 2);
807  return kTRUE;
808  }
809 
810  // Add to G3toG4, VGM
811  if (solid->GetEntityType() == "G4Hype") {
812  shapeType = "HYPE";
813  npar = 5;
814  par.Set(npar);
815  G4Hype* hype = (G4Hype*)solid;
816  par.AddAt(hype->GetInnerRadius() / cm, 0);
817  par.AddAt(hype->GetOuterRadius() / cm, 1);
818  par.AddAt(hype->GetZHalfLength() / cm, 2);
819  par.AddAt(hype->GetInnerStereo(), 3); // ?? unit
820  par.AddAt(hype->GetOuterStereo(), 4); // ?? unit
821  return kTRUE;
822  }
823 
824  // Add G4Orb
825 
826  if (solid->GetEntityType() == "G4Para") {
827  shapeType = "PARA";
828  npar = 6;
829  par.Set(npar);
830  G4Para* para = (G4Para*)solid;
831  par.AddAt(para->GetXHalfLength() / cm, 0);
