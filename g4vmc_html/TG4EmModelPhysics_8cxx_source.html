124 void TG4EmModelPhysics::AddModel(TG4EmModel emModel,
125  const G4ParticleDefinition* particle, const G4String& regions)
126 {
129 
130  if (!particle->GetProcessManager()) {
131  TString message;
132  message = "Cannot add EM model to ";
133  message = particle->GetParticleName();
134  message += " : particle has not defined process manager";
135  TG4Globals::Warning("TG4EmModelPhysics", "AddMOdel", message);
136  return;
137  }
138 
139  // Get process name
140  G4ProcessVector* processVector =
141  particle->GetProcessManager()->GetProcessList();
142  for (G4int i = 0; i < processVector->length(); i++) {
143  G4String processName;
144  G4String currentProcessName = (*processVector)[i]->GetProcessName();
145 
146  // PAI applied to ionisation
147  if (currentProcessName.contains("Ioni") &&
148  (emModel == kPAIModel || emModel == kPAIPhotonModel)) {
149  processName = currentProcessName;
150  }
151 
152  // UrbanMsc applied to msc
153  if (currentProcessName.contains("msc") &&
154  (emModel == kSpecialUrbanMscModel)) {
155  processName = currentProcessName;
156  }
157 
158  if (!processName.size()) continue;
159 
160  // CreateEM model
161  //
162  G4VEmModel* g4EmModel = 0;
163  G4VEmFluctuationModel* g4FluctModel = 0;
164  if (emModel == kPAIModel) {
165  // PAI
166  G4PAIModel* pai = new G4PAIModel(particle, "PAIModel");
167  if (verboseLevel > 2) {
168  G4cout << "New G4PAIModel" << G4endl;
169  }
170  g4EmModel = pai;
171  g4FluctModel = pai;
172  }
173  else if (emModel == kPAIPhotonModel) {
174  // PAIPhoton
175  if (verboseLevel > 2) {
176  G4cout << "New G4PAIPhotModel" << G4endl;
177  }
178  G4PAIPhotModel* paiPhot = new G4PAIPhotModel(particle, "PAIPhotModel");
179  g4EmModel = paiPhot;
180  g4FluctModel = paiPhot;
181  }
182  else if (emModel == kSpecialUrbanMscModel) {
183  // SpecialUrbanMsc
184  if (verboseLevel > 2) {
185  G4cout << "New TG4SpecialUrbanMscModel" << G4endl;
186  }
187  g4EmModel = new TG4SpecialUrbanMscModel();
188  g4FluctModel = 0;
189  }
190 
191  // Get regions
192  std::vector<G4String> regionVector;
193  if (regions.size()) {
194  // use analysis utility to tokenize regions
195  G4Analysis::Tokenize(regions, regionVector);
196  }
197  else {
198  // Get world default region
199  G4LogicalVolume* worldLV =
200  TG4GeometryServices::Instance()->GetWorld()->GetLogicalVolume();
201  regionVector.push_back(worldLV->GetRegion()->GetName());
202  }
203 
204  for (G4int j = 0; j < G4int(regionVector.size()); ++j) {
205 
206  G4String regionName = regionVector[j];
207 
208  if (VerboseLevel() > 2) {
209  G4cout << "Adding EM model: " << GetEmModelName(emModel)
210  << " to particle: " << particle->GetParticleName()
211  << " process: " << processName
212  << " region(=material): " << regionName << G4endl;
213  }
214 
215  G4LossTableManager::Instance()->EmConfigurator()->SetExtraEmModel(
216  particle->GetParticleName(), processName, g4EmModel, regionName, 0.0,
217  DBL_MAX, g4FluctModel);
218  }
219  }
220 }

223 void TG4EmModelPhysics::AddModels(
224  const std::vector<TG4ModelConfiguration*>& models)
225 {
228 
229  if (VerboseLevel() > 1) {
230  G4cout << "TG4EmModelPhysics::AddModels" << G4endl;
231  std::vector<TG4ModelConfiguration*>::const_iterator it;
232  for (it = models.begin(); it != models.end(); it++) {
233  (*it)->Print();
234  }
235  }
236 
237  std::vector<TG4ModelConfiguration*>::const_iterator it;
238  for (it = models.begin(); it != models.end(); it++) {
239 
240  // Get model configuration
241  TG4EmModel emModel = GetEmModel((*it)->GetModelName());
242  G4String particles = (*it)->GetParticles();
243  G4String regions = (*it)->GetRegions();
244 
245  if (!regions.size()) {
246  // add warning
247  TString message;
248  message = "No regions are defined for ";
249  message += (*it)->GetModelName().data();
250  TG4Globals::Warning("TG4EmModelPhysics", "AddModels", message);
251  continue;
252  }
253 
254  // Add selected models
255  auto aParticleIterator = GetParticleIterator();
256  aParticleIterator->reset();
257  while ((*aParticleIterator)()) {
258  G4ParticleDefinition* particle = aParticleIterator->value();
259  G4String particleName = particle->GetParticleName();
260 
261  // skip particles which are not in selection
262  if (particles != "all" &&
263  particles.find(particle->GetParticleName()) == std::string::npos) {
264  continue;
265  }
266 
267  // skip also monopole (experimental)
268  if (particle->GetParticleName() == "monopole") {
269  G4cout << "TG4EmModelPhysics::AddModels - skipping monopole" << G4endl;
270  continue;
271  }
272 
273  AddModel(emModel, particle, regions);
274  }
275  }
