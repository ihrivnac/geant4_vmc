68 void TG4FastSimulationPhysics::UpdateRegions(
69  const std::vector<TG4ModelConfiguration*>& models)
70 {
71 
72  if (VerboseLevel() > 1) {
73  G4cout << "TG4FastSimulationPhysics::UpdateRegions" << G4endl;
74  }
75 
76  // Process fast models configurations
77  std::vector<TG4ModelConfiguration*>::const_iterator it;
78  for (it = models.begin(); it != models.end(); it++) {
79 
80  // Get model configuration
81  G4String modelName = (*it)->GetModelName();
82  G4String regions = (*it)->GetRegions();
83  G4VFastSimulationModel* fastSimulationModel =
84  (*it)->GetFastSimulationModel();
85 
86  if (VerboseLevel() > 1) {
87  G4cout << "Adding fast simulation model " << modelName << " to regions "
88  << regions << G4endl;
89  }
90 
91  if (!fastSimulationModel) {
92  TString text = "The fast simulation model ";
93  text += modelName.data();
94  text += " was not found.";
95  TG4Globals::Warning("TG4FastSimulationPhysics", "UpdateRegions", text);
96  continue;
97  }
98 
99  if (!regions.size()) {
100  TString text = "The fast simulation model ";
101  text += modelName.data();
102  text += " has no associated region.";
103  TG4Globals::Warning("TG4FastSimulationPhysics", "UpdateRegions", text);
104  continue;
105  }
106 
107  std::vector<G4String> regionVector;
108  // use analysis utility to tokenize regions
109  G4Analysis::Tokenize(regions, regionVector);
110 
111  for (G4int j = 0; j < G4int(regionVector.size()); ++j) {
112 
113  // Get region
114  G4Region* region =
115  G4RegionStore::GetInstance()->GetRegion(regionVector[j]);
116 
117  if (!region) {
118  TString text = "The region ";
119  text += regionVector[j].data();
120  text += " was not found.";
121  TG4Globals::Warning("TG4FastSimulationPhysics", "UpdateRegions", text);
122  continue;
123  }
124 
125  // Retrieve fast simulation manager ou create one if needed.
126  G4FastSimulationManager* fastSimulationManager =
127  region->GetFastSimulationManager();
128  if (!fastSimulationManager) {
129  if (VerboseLevel() > 1) {
130  G4cout << "creating G4FastSimulationManager for the region" << G4endl;
131  }
132  // TO DO: CHECK THIS
133  G4bool isUnique = false;
134  fastSimulationManager = new G4FastSimulationManager(region, isUnique);
135  }
136  fastSimulationManager->AddFastSimulationModel(fastSimulationModel);
137  }

143 TG4FastSimulationPhysics::GetOrCreateFastSimulationProcess(
144  const G4String& modelName)
145 {
148 
149  // Create fast simulation process map if it does not yet exist
150  if (!fgProcessMap) {
151  fgProcessMap = new ProcessMap();
152  }
153 
154  // Create one thread-local fast simulation process per model
155  G4FastSimulationManagerProcess* fastSimulationProcess = 0;
156  ProcessMap::iterator it = fgProcessMap->find(modelName);
157  if (it != fgProcessMap->end()) {
158  fastSimulationProcess = it->second;
159  }
160  else {
161  // fastSimulationProcess = new G4FastSimulationManagerProcess(modelName);

164  G4cout << "... created fastSimulationProcess" << G4endl;
165  }

171 void TG4FastSimulationPhysics::AddFastSimulationProcess(
172  const std::vector<TG4ModelConfiguration*>& models)
173 {
176 
177  if (VerboseLevel() > 1) {
178  G4cout << "TG4FastSimulationPhysics::AddFastSimulationProcess" << G4endl;
179  }
180 
181  // Process fast models configurations
182  std::vector<TG4ModelConfiguration*>::const_iterator it;
183  for (it = models.begin(); it != models.end(); it++) {
184 
185  // Get model name
186  G4String modelName = (*it)->GetModelName();
187  G4String particles = (*it)->GetParticles();
188 
189  // Get or create fast simulation process
190  G4FastSimulationManagerProcess* fastSimulationProcess =
191  GetOrCreateFastSimulationProcess(modelName);
192 
193  // Add fast simulation process to selected particles
194  auto aParticleIterator = GetParticleIterator();
195  aParticleIterator->reset();
196  while ((*aParticleIterator)()) {
197  G4ParticleDefinition* particle = aParticleIterator->value();
198  G4String particleName = particle->GetParticleName();
199 
200  // skip particles which are not in selection
201  if (particles != "all" && (!(*it)->HasParticle(particleName))) {
202  continue;
203  }
204 
205  // skip particles which do not have process manager
206  if (!particle->GetProcessManager()) continue;
207 
208  if (VerboseLevel() > 2) {
209  G4cout << "Adding model " << modelName << " to particle "
210  << particle->GetParticleName() << G4endl;
211  }
212 
213  // Set the process to the particle process manager
214  particle->GetProcessManager()->AddDiscreteProcess(fastSimulationProcess);
215  }
216  }
217 }

230 void TG4FastSimulationPhysics::ConstructProcess()
231 {
232  if (VerboseLevel() > 1) {
233  G4cout << "TG4FastSimulationPhysics::ConstructProcess " << G4endl;
234  }
235 
236  // Get model configurations vector from geometry manager
237  TG4ModelConfigurationManager* fastModelsManager =
238  TG4GeometryManager::Instance()->GetFastModelsManager();
239 
240  const std::vector<TG4ModelConfiguration*>& models =
241  fastModelsManager->GetVector();
242 
243  // Do nothing if no models were set
244  if (models.size() == 0) {
245  if (VerboseLevel() > 1) {
246  G4cout << "No fast simulation models are defined." << G4endl;
247  }
248  return;
249  }
250 
251  // Construct user models
252  if (fUserFastSimulation) {
253  fUserFastSimulation->Construct();
254  }
255 
256  // Update regions
257  UpdateRegions(models);
258 
259  // Add fast simulation process to particles
260  AddFastSimulationProcess(models);
261 
262  if (VerboseLevel() > 0) {
263  G4cout << "### Fast simulation models added to physics processes" << G4endl;
264  }
265 }
