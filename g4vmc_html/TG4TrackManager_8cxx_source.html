106  TG4TrackInformation* trackInfo = GetTrackInformation(track);

108  if (!trackInfo) {

112  fG4TrackingManager->SetUserTrackInformation(trackInfo);

128  else {

130  trackIndex = fMCStack->GetNtrack();
131  if (overWrite) trackIndex--;
132  }
133  else
134  trackIndex = fTrackCounter;
135  // if secondaries are not stacked in VMC stack
136  // use own counter for setting track index
137  }
138  if (VerboseLevel() > 1)
139  G4cout << "TG4TrackManager::SetTrackInformation: setting " << trackIndex
140  << G4endl;
141 
142  trackInfo->SetTrackParticleID(trackIndex);
143  }
144 
145  // set current track number
146  // fMCStack->SetCurrentTrack(trackIndex);
147  ++fTrackCounter;
148 
149  return trackIndex;

165  const G4TrackVector* secondaryTracks =

176 #ifdef MCDEBUG

182 #endif
183  G4int parentParticleID = parentInfo->GetTrackParticleID();
184 
185  // get or create track information and set it to the G4Track
186  TG4TrackInformation* trackInfo = GetTrackInformation(secondary);

191  // the track information is deleted together with its
192  // G4Track object
193  }

195  secondary->SetUserInformation(trackInfo);

218 //_____________________________________________________________________________
219 void TG4TrackManager::TrackToStack(const G4Track* track, G4bool /*overWrite*/)
220 #endif
221 {
224 
225  if (VerboseLevel() > 2) G4cout << "TG4TrackManager::TrackToStack" << G4endl;
226 
227  // parent particle index
228  G4int parentID = track->GetParentID();
229  G4int motherIndex;
230  if (parentID == 0) {
231  motherIndex = -1;
232  }
233  else {
234  motherIndex = GetTrackInformation(track)->GetParentParticleID();
235  }
236 
237  // PDG code
238  G4int pdg =
239  TG4ParticlesManager::Instance()->GetPDGEncoding(track->GetDefinition());
240 
241  // track kinematics
242  G4ThreeVector momentum = track->GetMomentum();
243  momentum *= 1. / (TG4G3Units::Energy());
244 
245  G4double px = momentum.x();
246  G4double py = momentum.y();
247  G4double pz = momentum.z();
248  G4double e = track->GetTotalEnergy() / TG4G3Units::Energy();
249 
250  G4ThreeVector position = track->GetPosition();
251  position *= 1. / (TG4G3Units::Length());
252  G4double vx = position.x();
253  G4double vy = position.y();
254  G4double vz = position.z();
255  G4double t = track->GetGlobalTime() / TG4G3Units::Time();
256 
257  G4ThreeVector polarization = track->GetPolarization();
258  G4double polX = polarization.x();
259  G4double polY = polarization.y();
260  G4double polZ = polarization.z();
261 
262  // production process
263  TMCProcess mcProcess;
264  const G4VProcess* kpProcess = track->GetCreatorProcess();
265  if (!kpProcess) {
266  mcProcess = kPPrimary;
267  }
268  else {
269  mcProcess = TG4PhysicsManager::Instance()->GetMCProcess(kpProcess);
270  // distinguish kPDeltaRay from kPEnergyLoss
271  if (mcProcess == kPEnergyLoss) mcProcess = kPDeltaRay;
272  }
273 
274  G4double weight = track->GetWeight();
275 
276  G4int status = 0;
277  if (fSaveDynamicCharge) {
278  // Store the dynamic particle charge (which in case of ion may
279  // be different from PDG charge) as status as there is no other
280  // place where we can do it
281  status = G4int(track->GetDynamicParticle()->GetCharge() / eplus);
282  }
283 
284  G4int ntr;
285 #ifdef STACK_WITH_KEEP_FLAG
286  // create particle
287  fMCStack->PushTrack(0, motherIndex, pdg, px, py, pz, e, vx, vy, vz, t, polX,
288  polY, polZ, mcProcess, ntr, weight, status, overWrite);
289  // Experimental code with flagging tracks in stack for overwrite;
290  // not yet available in distribution
291 #else
292  fMCStack->PushTrack(0, motherIndex, pdg, px, py, pz, e, vx, vy, vz, t, polX,
293  polY, polZ, mcProcess, ntr, weight, status);
294 #endif
295 }

353 void TG4TrackManager::SaveSecondaries(
354  const G4Track* track, const G4TrackVector* secondaries)
355 {
357 
358  if (track->GetTrackID() != fCurrentTrackID) {
359  fCurrentTrackID = track->GetTrackID();
360  fNofSavedSecondaries = 0;
361  }
362 
363  // Store parent track Id
364  SetParentToTrackInformation(track);
365 
366  for (G4int i = fNofSavedSecondaries; i < G4int(secondaries->size()); ++i) {
367 
368  G4Track* secondary = (*secondaries)[i];
369 
370  // G4cout << i << "th secondary to be saved: "
371  // << secondary->GetDefinition()->GetParticleName()
372  // << G4endl;
373 
374  if (GetTrackInformation(secondary) &&
375  GetTrackInformation(secondary)->IsUserTrack())
376  return;
377 
378  // Set track Id
379  SetTrackInformation(secondary);
380 
381  // Save track in stack
382  TrackToStack(secondary);
383 
384  // Notify a stack popper (if activated) about saving this secondary
385  if (fStackPopper) fStackPopper->Notify();
386  ++fNofSavedSecondaries;
387  }
388 }
