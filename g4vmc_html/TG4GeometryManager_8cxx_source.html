618 void TG4GeometryManager::ConstructZeroFields()
619 {
623 
624  if (VerboseLevel() > 1)
625  G4cout << "TG4GeometryManager::ConstructZeroFields()" << G4endl;
626 
627  G4bool forceToAllDaughters = false;
628 
629  // Set a dummy field manager to all LV first in order to avoid propagation
630  // of zero field to volume daughters
631  G4FieldManager* dummyFieldManager = new G4FieldManager();
632  G4LogicalVolumeStore* lvStore = G4LogicalVolumeStore::GetInstance();
633  for (G4int i = 0; i < G4int(lvStore->size()); i++) {
634  G4LogicalVolume* lv = (*lvStore)[i];
635  lv->SetFieldManager(dummyFieldManager, forceToAllDaughters);
636  }
637 
638  // Set zero field manager to volumes associated with a tracking medium
639  // with ifield value = 0.
640  G4FieldManager* fieldManager = 0;
641  for (G4int i = 0; i < G4int(lvStore->size()); i++) {
642 
643  G4LogicalVolume* lv = (*lvStore)[i];
644 
645  // skip volume without medium
646  TG4Medium* medium =
647  TG4GeometryServices::Instance()->GetMediumMap()->GetMedium(lv, false);
648  if (!medium) continue;
649 
650  // Skip volumes with ifield != 0
651  if (medium->GetIfield() != 0) {
652  if (VerboseLevel() > 1) {
653  G4cout << "Global field in logical volume: " << lv->GetName() << G4endl;
654  }
655  continue;
656  }
657 
658  // create field manager if it does not exist yet
659  if (!fieldManager) {
660  fieldManager = new G4FieldManager();
661  // CHECK if we need to delete it
662  fieldManager->SetDetectorField(0);
663  fieldManager->CreateChordFinder(0);
664  }
665  lv->SetFieldManager(fieldManager, forceToAllDaughters);
666 
667  if (VerboseLevel() > 1) {
668  G4cout << "Zero magnetic field set to logical volume: " << lv->GetName()
669  << G4endl;
670  }
671  }
672 
673  // Remove the dummy field manager to all LV without zero field
674  for (G4int i = 0; i < G4int(lvStore->size()); i++) {
675  G4LogicalVolume* lv = (*lvStore)[i];
676 
677  if (lv->GetFieldManager() == dummyFieldManager) {
678  lv->SetFieldManager(0, forceToAllDaughters);
679  }
680  }
681 
682  // Delete the dummy field manager
683  delete dummyFieldManager;
684 }

896 void TG4GeometryManager::SetUserLimits(
897  const TG4G3CutVector& cuts, const TG4G3ControlVector& controls) const
898 {
900 
901  if (VerboseLevel() > 1)
902  G4cout << "TG4GeometryManager::SetUserLimits" << G4endl;
903 
904  G4LogicalVolumeStore* lvStore = G4LogicalVolumeStore::GetInstance();
905 
906  for (G4int i = 0; i < G4int(lvStore->size()); i++) {
907  G4LogicalVolume* lv = (*lvStore)[i];
908  TG4Medium* medium = fGeometryServices->GetMediumMap()->GetMedium(lv, false);
909 
910  if (!medium) continue;
911 
912  // get limits if already exist
913  TG4Limits* tg4Limits = 0;
914  G4UserLimits* limits = medium->GetLimits();
915  tg4Limits = fGeometryServices->GetLimits(limits, cuts, controls);
916 
917  // get tracking medium name
918  G4String name = medium->GetName();
919 
920  if (tg4Limits) {
921  tg4Limits->SetName(name);
922  }
923  else {
924  // Check if the step below is needed
925  tg4Limits = fGeometryServices->FindLimits2(name, true);
926  if (!tg4Limits) {
927  tg4Limits = new TG4Limits(name, cuts, controls);
928  }
929  }
930 
931  // set new limits back to medium
932  medium->SetLimits(tg4Limits);
933 
934  // inactivate max step defined by user
935  // if its activation was not asked explicitely
936  if (!fIsUserMaxStep) tg4Limits->SetMaxAllowedStep(DBL_MAX);
937 
938  // limit max step for low density materials (< AIR)
939  if (fIsMaxStepInLowDensityMaterials &&
940  lv->GetMaterial()->GetDensity() < fLimitDensity)
941  tg4Limits->SetMaxAllowedStep(fMaxStepInLowDensityMaterials);
942 
943  // set max step the default value
944  tg4Limits->SetDefaultMaxAllowedStep();
945 
946  // update controls in limits according to the setup
947  // in the passed vector
948  tg4Limits->Update(controls);
949 
950  // set limits to logical volume
951  lv->SetUserLimits(tg4Limits);
952  }
953 
954  if (VerboseLevel() > 1)
955  G4cout << "TG4GeometryManager::SetUserLimits done" << G4endl;
956 }
