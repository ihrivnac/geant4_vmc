78 Double_t TG4OpGeometryManager::AddUnit(
79  const G4String& propertyName, Double_t value) const
80 {
83 
84  if (propertyName == "ABSLENGTH" || propertyName == "MIEHG" ||
85  propertyName == "RAYLEIGH" || propertyName == "WLSABSLENGTH") {
86  return value * TG4G3Units::Length();
87  }
88 
89  if (propertyName == "BIRKS_CONSTANT") {
90  return value * TG4G3Units::Length() / TG4G3Units::Energy();
91  }
92 
93  if (propertyName == "SCINTILLATIONYIELD") {
94  return value / TG4G3Units::Energy();
95  }
96 
97  if (propertyName == "FASTTIMECONSTANT" ||
98  propertyName == "FASTSCINTILLATIONRISETIME" ||
99  propertyName == "SLOWTIMECONSTANT" ||
100  propertyName == "SLOWSCINTILLATIONRISETIME" ||
101  propertyName == "WLSTIMECONSTANT") {
102  return value * TG4G3Units::Time();
103  }
104 
105  if (propertyName == "GROUPVEL") {
106  // use the same convention as in Root for velocity of light
107  return value * m / s;
108  }
109 
110  if (propertyName == "BACKSCATTERCONSTANT" || propertyName == "EFFICIENCY" ||
111  propertyName == "FASTCOMPONENT" || propertyName == "IMAGINARYRINDEX" ||
112  propertyName == "MIEHG_BACKWARD" || propertyName == "MIEHG_FORWARD" ||
113  propertyName == "MIEHG_FORWARD_RATIO" || propertyName == "REALRINDEX" ||
114  propertyName == "REFLECTIVITY" || propertyName == "RESOLUTIONSCALE" ||
115  propertyName == "RINDEX" || propertyName == "SLOWCOMPONENT" ||
116  propertyName == "SPECULARLOBECONSTANT" ||
117  propertyName == "SPECULARSPIKECONSTANT" ||
118  propertyName == "TRANSMITTANCE" || propertyName == "WLSCOMPONENT" ||
119  propertyName == "WLSMEANNUMBERPHOTONS" || propertyName == "YIELDRATIO" ||
120  propertyName == "PROTONSCINTILLATIONYIELD" ||
121  propertyName == "DEUTERONSCINTILLATIONYIELD" ||
122  propertyName == "TRITONSCINTILLATIONYIELD" ||
123  propertyName == "ALPHASCINTILLATIONYIELD" ||
124  propertyName == "IONSCINTILLATIONYIELD" ||
125  propertyName == "ELECTRONSCINTILLATIONYIELD") {
126 
127  return value;
128  }
129 
130  TString text = "Unknown material property ";
131  text += propertyName.data();
132  text += "x\n";
133  text += "No units conversion performed.";
134  TG4Globals::Warning("TG4OpGeometryManager", "AddUnit", text);
135  return value;
136 }

178 void TG4OpGeometryManager::SetCerenkov(Int_t itmed, Int_t npckov,
179  Double_t* ppckov, Double_t* absco, Double_t* effic, Double_t* rindex)
180 {
198 
199  // add units and detect if Rindex == 0
200  G4bool isRindex = false;
201  G4double* ppckov2 = fGeometryServices->CreateG4doubleArray(ppckov, npckov);
202  G4double* absco2 = fGeometryServices->CreateG4doubleArray(absco, npckov);
203  for (G4int i = 0; i < npckov; i++) {
204  ppckov2[i] *= TG4G3Units::Energy();
205  absco2[i] *= TG4G3Units::Length();
206  if (rindex[i] != 0.0) isRindex = true;
207  }
208 
209  // get material of medium from table
210  TG4Medium* medium = fGeometryServices->GetMediumMap()->GetMedium(itmed);
211  if (!medium) {
212  TString text = "Medium ";
213  text += itmed;
214  text += " not found.";
215  TG4Globals::Warning("TG4OpGeometryManager", "SetCerenkov", text);
216  delete[] ppckov2;
217  delete[] absco2;
218  return;
219  }
220  G4Material* material = medium->GetMaterial();
221 
222  // get materials properties table or create it if it does not yet
223  // exist
224  G4MaterialPropertiesTable* table = material->GetMaterialPropertiesTable();
225  if (!table) {
226  table = new G4MaterialPropertiesTable();
227  material->SetMaterialPropertiesTable(table);
228  }
229 
230  // add material properties into the table
231  table->AddProperty("ABSLENGTH", ppckov2, absco2, npckov);
232  // used in G4OpAbsorption process
233  table->AddProperty("EFFICIENCY", ppckov2, effic, npckov);
234  // used in G4OpBoundary process
235 
236  // Set RINDEX only if defined with non zero values.
237  // The zero value is a Geant3 convention to define material as metal which was
238  // not adopted in Geant4, where 0 values cause unpredicatable behavior in
239  // G4OpBoundary process
240  if (isRindex) {
241  table->AddProperty("RINDEX", ppckov2, rindex, npckov);
242  // used in G4Cerenkov, G4OpRayleigh, G4OpBoundary
243  }
244 
245  delete[] ppckov2;
246  delete[] absco2;
247 
248  // verbose
249  if (VerboseLevel() > 0) {
250  G4cout << "The tables for UV photon tracking set for "
251  << material->GetName() << G4endl;
252  }
253 }
