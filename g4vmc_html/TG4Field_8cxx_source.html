106 G4EquationOfMotion* TG4Field::CreateEquation(EquationType equation)
107 {
109 
110  // magnetic fields
111  G4MagneticField* magField = 0;
112  if (equation == kMagUsualEqRhs || equation == kMagSpinEqRhs) {
113  magField = dynamic_cast<G4MagneticField*>(fG4Field);
114  if (!magField) {
115  // add warning
116  return 0;
117  }
118  }
119 
120  // electromagnetic fields
121  G4ElectroMagneticField* elMagField = 0;
122  if (equation >= kEqMagElectric && equation <= kEqEMFieldWithEDM) {
123  elMagField = dynamic_cast<G4ElectroMagneticField*>(fG4Field);
124  if (!elMagField) {
125  // add warning
126  return 0;
127  }
128  }
129 
130  // electromagnetic fields
131  switch (equation) {
132  case kMagUsualEqRhs:
133  return new G4Mag_UsualEqRhs(magField);
134  break;
135 
136  case kMagSpinEqRhs:
137  return new G4Mag_SpinEqRhs(magField);
138  break;
139 
140  case kEqMagElectric:
141  return new G4EqMagElectricField(elMagField);
142  break;
143 
144  case kEqEMFieldWithSpin:
145  return new G4EqEMFieldWithSpin(elMagField);
146  break;
147 
148  case kEqEMFieldWithEDM:
149  return new G4EqEMFieldWithEDM(elMagField);
150  break;
151  case kUserEquation:
152  // nothing to be done
153  return 0;
154  break;
155  }
156 
157  TG4Globals::Exception(
158  "TG4Field", "CreateEquation:", "Unknown equation type.");
159  return 0;
160 }

251 void TG4Field::Update(const TG4FieldParameters& parameters)
252 {
254 
255  // Create field
256  CreateG4Field(parameters, fVirtualMagField);

259 
260  if (!fLogicalVolume) {

272

274  if (parameters.GetEquationType() == kUserEquation) {
275  fEquation = parameters.GetUserEquationOfMotion();
276  fEquation->SetFieldObj(fG4Field);
277  }
278  else {

280  }
281

283  if (parameters.GetStepperType() == kUserStepper) {
284  fStepper = parameters.GetUserStepper();
285  }
286  else {
287  fStepper = CreateStepper(fEquation, parameters.GetStepperType());
288  }
289 
290  // Chord finder
291  G4ChordFinder* chordFinder = 0;
292  if (parameters.GetFieldType() == kMagnetic) {
293  // Chord finder
294  chordFinder = new G4ChordFinder(static_cast<G4MagneticField*>(fG4Field),
295  parameters.GetStepMinimum(), fStepper);
296  chordFinder->SetDeltaChord(parameters.GetDeltaChord());
297  }
298  else if (parameters.GetFieldType() == kElectroMagnetic) {
299  G4MagInt_Driver* intDriver = new G4MagInt_Driver(
300  parameters.GetStepMinimum(), fStepper, fStepper->GetNumberOfVariables());
301  if (intDriver) {
302  // Chord finder
303  chordFinder = new G4ChordFinder(intDriver);
304  }
305  }
306  fieldManager->SetChordFinder(chordFinder);
307 
308  fieldManager->SetMinimumEpsilonStep(parameters.GetMinimumEpsilonStep());
309  fieldManager->SetMaximumEpsilonStep(parameters.GetMaximumEpsilonStep());
310  fieldManager->SetDeltaOneStep(parameters.GetDeltaOneStep());
311  fieldManager->SetDeltaIntersection(parameters.GetDeltaIntersection());
312 }
