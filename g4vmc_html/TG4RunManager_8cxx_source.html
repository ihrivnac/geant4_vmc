173 void TG4RunManager::ConfigureRunManager()
174 {
177 
178  // Geometry construction and navigator
179  //
180  if (VerboseLevel() > 1)
181  G4cout << "TG4RunManager::ConfigureRunManager " << this << G4endl;
182 
183  TString userGeometry = fRunConfiguration->GetUserGeometry();
184 
185  TG4GeometryManager::Instance()->SetUserRegionConstruction(
186  fRunConfiguration->CreateUserRegionConstruction());
187 
188  TG4GeometryManager::Instance()->SetUserPostDetConstruction(
189  fRunConfiguration->CreateUserPostDetConstruction());
190 
191  // Root navigator
192 #ifdef USE_G4ROOT
193  TG4RootNavMgr* rootNavMgr = 0;
194  if (userGeometry == "VMCtoRoot" || userGeometry == "Root") {
195 
196  // Construct geometry via VMC application
197  if (TG4GeometryManager::Instance()->VerboseLevel() > 0)
198  G4cout << "Running TVirtualMCApplication::ConstructGeometry";
199 
200  TG4StateManager::Instance()->SetNewState(kConstructGeometry);
201  TVirtualMCApplication::Instance()->ConstructGeometry();
202  TG4StateManager::Instance()->SetNewState(kNotInApplication);
203 
204  // Set top volume and close Root geometry if not yet done
205  if (!gGeoManager->IsClosed()) {
206  TGeoVolume* top = (TGeoVolume*)gGeoManager->GetListOfVolumes()->First();
207  gGeoManager->SetTopVolume(top);
208  gGeoManager->CloseGeometry();
209  }
210 
211  // Now that we have the ideal geometry, call application misalignment code
212  TG4StateManager::Instance()->SetNewState(kMisalignGeometry);
213  TVirtualMCApplication::Instance()->MisalignGeometry();
214  TG4StateManager::Instance()->SetNewState(kNotInApplication);
215 
216  // Pass geometry to G4Root navigator
217  rootNavMgr = TG4RootNavMgr::GetInstance(gGeoManager);
218  }
219 #endif
220 
221  // G4 run manager
222 #ifdef G4MULTITHREADED
223  if (fRunConfiguration->IsMTApplication()) {
224  fRunManager = new G4MTRunManager();
225  fRunManager->SetUserInitialization(new TG4WorkerInitialization());
226  }
227  else {
228  fRunManager = new G4RunManager();
229  }
230 #else
231  fRunManager = new G4RunManager();
232 #endif
233  if (VerboseLevel() > 1) {
234  G4cout << "G4RunManager has been created." << G4endl;
235  }
236 
237  if (userGeometry != "VMCtoRoot" && userGeometry != "Root") {
238  fRunManager->SetUserInitialization(
239  fRunConfiguration->CreateDetectorConstruction());
240  if (VerboseLevel() > 1)
241  G4cout << "CreateDetectorConstruction done." << G4endl;
242  }
243  else {
244 #ifdef USE_G4ROOT
245  G4int nthreads = 1;
246 #ifdef G4MULTITHREADED
247  if (fRunConfiguration->IsMTApplication()) {
248  nthreads = G4MTRunManager::GetMasterRunManager()->GetNumberOfThreads();
249  }
250 #endif
251  rootNavMgr->Initialize(new TG4PostDetConstruction(), nthreads);
252  rootNavMgr->ConnectToG4();
253 #else
254  TG4Globals::Exception("TG4RunManager", "ConfigureRunManagerTG4MCGeometry",
255  "geomVMCtoRoot and geomRoot options require Geant4 VMC built with "
256  "G4Root.");
257 #endif
258  }
259 
260  // Other mandatory classes
261  //
262  fRunManager->SetUserInitialization(fRunConfiguration->CreatePhysicsList());
263  if (VerboseLevel() > 1) G4cout << "CreatePhysicsList done." << G4endl;
264 
265  // User fast simulation
266  TG4VUserFastSimulation* userFastSimulation =
267  fRunConfiguration->CreateUserFastSimulation();
268  if (userFastSimulation) {
269  if (TG4SpecialPhysicsList::Instance()) {
270  TG4SpecialPhysicsList::Instance()->SetUserFastSimulation(
271  userFastSimulation);
272  }
273  else {
274  TG4Globals::Warning("TG4RunManager", "ConfigureRunManager",
275  "TG4SpecialPhysicsList must be instantiated to use fast simulation");
276  }
277  }
278 
279  fRunManager->SetUserInitialization(
280  new TG4ActionInitialization(fRunConfiguration));
281  if (VerboseLevel() > 1)
282  G4cout << "Create ActionInitialization done." << G4endl;
283 
284  // Regions manager
285  //
286  fRegionsManager = new TG4RegionsManager();
287 
288  if (VerboseLevel() > 1)
289  G4cout << "TG4RunManager::ConfigureRunManager done " << this << G4endl;
290 }

358 {
361

363 
364  G4bool isArg = false;
365  for (G4int i = 0; i < fARGC; i++) {
366  if (G4String(fARGV[i]) == arg) isArg = true;
367  if (isArg && i + 1 < fARGC) fARGV[i] = fARGV[i + 1];
368  }
369
