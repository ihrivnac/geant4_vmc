63 void TG4ComposedPhysicsList::ApplyGammaToMuonsCrossSectionFactor()
64 {
67 
68  if (fGammaToMuonsCrossSectionFactor > 0.) {
69  G4ParticleDefinition* gamma = G4Gamma::Gamma();
70  G4ProcessManager* processManager = gamma->GetProcessManager();
71  G4ProcessVector* processVector = processManager->GetProcessList();
72  G4bool done = false;
73  // get G4GammaConversionToMuons
74  for (G4int i = 0; i < processVector->length(); i++) {
75  G4GammaConversionToMuons* gammaToMuMu =
76  dynamic_cast<G4GammaConversionToMuons*>((*processVector)[i]);
77  if (gammaToMuMu) {
78  if (VerboseLevel() > 0) {
79  G4cout << "### Set gamma to muons cross section factor user value: "
80  << fGammaToMuonsCrossSectionFactor << G4endl;
81  }
82  gammaToMuMu->SetCrossSecFactor(fGammaToMuonsCrossSectionFactor);
83  done = true;
84  break;
85  }
86  }
87  if (!done) {
88  TG4Globals::Warning("TG4ComposedPhysicsList",
89  "ApplyGammaToMuonsCrossSectionFactor",
90  "Cannot set gamma to muons cross section factor, GammaToMuons must be "
91  "activated first.");
92  }
93  }
94 }

148 void TG4ComposedPhysicsList::SetCuts()
149 {
151 
152  if (VerboseLevel() > 1) G4cout << "TG4ComposedPhysicsList::SetCuts" << G4endl;
153 
154  if (fIsProductionCutsTableEnergyRange) {
155  G4ProductionCutsTable::GetProductionCutsTable()->SetEnergyRange(
156  fProductionCutsTableEnergyMin, fProductionCutsTableEnergyMax);
157  }
158 
159  // Get default range cut values from physics manager
160  G4double rangeCutGam = TG4PhysicsManager::Instance()->GetCutForGamma();
161  G4double rangeCutEle = TG4PhysicsManager::Instance()->GetCutForElectron();
162  G4double rangeCutPos = TG4PhysicsManager::Instance()->GetCutForPositron();
163  G4double rangeCutPro = TG4PhysicsManager::Instance()->GetCutForProton();
164 
165  // Set cut values for gamma at first and for e- second and next for e+,
166  // because some processes for e+/e- need cut values for gamma
167  SetCutValue(rangeCutGam, "gamma");
168  SetCutValue(rangeCutEle, "e-");
169  SetCutValue(rangeCutPos, "e+");
170  SetCutValue(rangeCutPro, "proton");
171 
172  // Set the default production cuts to all already created regions.
173  // The cuts can be then further customized via VMC cuts or
174  // via Geant4 UI commands called after G4RunManager initialization
175  // (in VMC examples in g4config2.in macros)
176 
177  // Loop over regions
178  G4RegionStore* regionStore = G4RegionStore::GetInstance();
179  for (G4int i = 0; i < G4int(regionStore->size()); i++) {
180  G4Region* region = (*regionStore)[i];
181 
182  // skip regions which already have production cuts defined
183  if (region->GetProductionCuts()) continue;
184 
185  G4ProductionCuts* cuts = new G4ProductionCuts();
186  cuts->SetProductionCut(rangeCutGam, 0);
187  cuts->SetProductionCut(rangeCutEle, 1);
188  cuts->SetProductionCut(rangeCutPos, 2);
189  cuts->SetProductionCut(rangeCutPro, 3);
190  region->SetProductionCuts(cuts);
191  }
192 
193  if (VerboseLevel() > 0) DumpCutValuesTable();
194 }
