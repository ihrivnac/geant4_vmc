41 TG4RunConfiguration::TG4RunConfiguration(const TString& userGeometry,
42  const TString& physicsList, const TString& specialProcess,
43  Bool_t specialStacking, Bool_t mtApplication)
44  : fUserGeometry(userGeometry),
45  fPhysicsListSelection(physicsList),
46  fSpecialProcessSelection(),
47  fSpecialStacking(specialStacking),
48  fMTApplication(mtApplication),
49  fSpecialControls(false),
50  fSpecialCuts(false),
51  fAGDDMessenger(0),
52  fGDMLMessenger(0),
53  fParameters()
54 {
56 
57  if (userGeometry != "geomVMCtoGeant4" && userGeometry != "geomVMCtoRoot" &&
58  userGeometry != "geomRoot" && userGeometry != "geomRootToGeant4" &&
59  userGeometry != "geomGeant4") {
60 
61  TG4Globals::Exception("TG4RunConfiguration", "TG4RunConfiguration",
62  "User geometry " + userGeometry + " not recognized." +
63  TG4Globals::Endl() + "Available options:" + TG4Globals::Endl() +
64  "geomVMCtoGeant4 geomVMCtoRoot geomRoot geomRootToGeant4 geomGeant4");
65  }
66 
67  G4int itoken = 0;
68  G4String token;
69  do {
70  token = TG4Globals::GetToken(itoken++, physicsList);
71 
72  if (!TG4EmPhysicsList::IsAvailableSelection(token) &&
73  !TG4HadronPhysicsList::IsAvailableSelection(token) &&
74  !TG4ExtraPhysicsList::IsAvailableSelection(token)) {
75 
76  TG4Globals::Exception("TG4RunConfiguration", "TG4RunConfiguration",
77  "Physics list selection " + physicsList + " not recognized." +
78  TG4Globals::Endl() + "Available options:" + TG4Globals::Endl() +
79  "EMonly, EMonly+Extra, Hadron_EM, Hadron_EM+Extra" +
80  TG4Globals::Endl() + " where EMonly = " +
81  TString(TG4EmPhysicsList::AvailableSelections()) +
82  TG4Globals::Endl() + " Hadron = " +
83  TString(TG4HadronPhysicsList::AvailableHadronSelections()) +
84  TG4Globals::Endl() + " EM = " +
85  TString(TG4HadronPhysicsList::AvailableEMSelections()) +
86  TG4Globals::Endl() + " Extra = " +
87  TString(TG4ExtraPhysicsList::AvailableSelections()) +
88  TG4Globals::Endl() +
89  " The Extra selections are cumulative, while Hadron selections are "
90  "exlusive." +
91  TG4Globals::Endl());
92  }
93  } while (token != "");
94 
95  G4String g4SpecialProcess(specialProcess.Data());
96 
97  if (g4SpecialProcess.contains("specialControls")) {
98  fSpecialControls = true;
99  // remove "specialControls" from the string passsed to special physics list
100  g4SpecialProcess.erase(g4SpecialProcess.find("specialControls"), 15);
101  if (g4SpecialProcess.find("++") != std::string::npos)
102  g4SpecialProcess.erase(g4SpecialProcess.find("++"), 1);
103  }
104  fSpecialProcessSelection = g4SpecialProcess;
105 
106  if (g4SpecialProcess.contains("specialCuts")) {
107  fSpecialCuts = true;
108  }
109 
110  if (!TG4SpecialPhysicsList::IsAvailableSelection(g4SpecialProcess)) {
111 
112  TG4Globals::Exception("TG4RunConfiguration", "TG4RunConfiguration",
113  "Special process selection " + specialProcess + " is not recognized." +
114  TG4Globals::Endl() + "Available options:" + TG4Globals::Endl() +
115  TString(TG4SpecialPhysicsList::AvailableSelections()));
116  }
117 
118 #ifdef USE_VGM
119  // instantiate XML messengers
120  fAGDDMessenger = new TG4VGMMessenger("AGDD", userGeometry.Data());
121  fGDMLMessenger = new TG4VGMMessenger("GDML", userGeometry.Data());
122 #endif
123 }

149 G4VUserPhysicsList* TG4RunConfiguration::CreatePhysicsList()
150 {
152 
153  TG4ComposedPhysicsList* builder = new TG4ComposedPhysicsList();
154 
155  // Decompose physics list selection
156  G4int itoken = 0;
157  G4String emSelection;
158  G4String hadronSelection;
159  G4String extraSelection;
160  G4String token;
161  G4bool isValid = true;
162  while (
163  (token = TG4Globals::GetToken(itoken++, fPhysicsListSelection)) != "") {
164  // The first token must be either EM or Hadronic physics list
165  if (itoken == 1) {
166  if (TG4EmPhysicsList::IsAvailableSelection(token)) {
167  emSelection = token;
168  }
169  else if (TG4HadronPhysicsList::IsAvailableSelection(token)) {
170  hadronSelection = token;
171  }
172  else {
173  isValid = false;
174  }
175  }
176  // The next tokens are Extra physics selections
177  else {
178  if (TG4ExtraPhysicsList::IsAvailableSelection(token)) {
179  extraSelection += token;
180  extraSelection += " ";
181  }
182  else {
183  isValid = false;
184  }
185  }
186  }
187 
188  if (!isValid) {
189  TG4Globals::Exception("TG4RunConfiguration", "TG4RunConfiguration",
190  "Physics list selection " + fPhysicsListSelection + " is not valid." +
191  TG4Globals::Endl() +
192  "The EMonly selections cannot be combined with Hadron selections." +
193  TG4Globals::Endl());
194  }
195 
196  if (emSelection != "") {
197  G4cout << "Adding EMPhysicsList " << emSelection << G4endl;
198  builder->AddPhysicsList(new TG4EmPhysicsList(emSelection));
199  }
200 
201  if (hadronSelection != "") {
202  G4cout << "Adding HadronPhysicsList " << hadronSelection << G4endl;
203  builder->AddPhysicsList(new TG4HadronPhysicsList(hadronSelection));
204  }
205 
206  if (extraSelection != "") {
207  G4cout << "Adding ExtraPhysicsList " << extraSelection << G4endl;
208  builder->AddPhysicsList(
209  new TG4ExtraPhysicsList(extraSelection, fParameters));
210  }
211 
212  // add option here
213  G4cout << "Adding SpecialPhysicsList " << fSpecialProcessSelection.Data()
214  << G4endl;
215  builder->AddPhysicsList(
216  new TG4SpecialPhysicsList(fSpecialProcessSelection.Data()));
217 
218  return builder;
219 }
