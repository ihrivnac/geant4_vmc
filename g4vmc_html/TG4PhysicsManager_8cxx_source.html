110 {
114 
115  // get medium from the map

118  if (!medium) {
119  // TG4GeometryServices::Instance()->GetMediumMap()->Print();
120  TString text = "mediumId=";
121  text += itmed;
122  TG4Globals::Warning(

142  medium->SetLimits(limits);
143 
144  // add units

252 void TG4PhysicsManager::SetSpecialControlsActivation()
253 {
256 
257  TG4G3PhysicsManager* g3PhysicsManager = TG4G3PhysicsManager::Instance();
258 
259  TG4G3ControlVector* controlVector = g3PhysicsManager->GetControlVector();
260  if (!controlVector) {
261  TG4Globals::Exception("TG4PhysicsManager", "SetSpecialControlsActivation",
262  "Vectors of processes controls is not set.");
263  return;
264  }
265 
266  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
267  for (G4int i = 0; i < G4int(particleTable->size()); ++i) {
268 
269  G4ParticleDefinition* particle = particleTable->GetParticle(i);
270  if (!particle) continue;
271 
272  G4ProcessManager* processManager = particle->GetProcessManager();
273  if (!processManager) continue;
274 
275  G4ProcessVector* processVector = processManager->GetProcessList();
276 
277  // activate or inactivate processes according to
278  // global setting in the control vector in G3 physics manager
279  //
280  for (G4int j = 0; j < processVector->length(); j++) {
281 
282  TG4G3ControlValue control =
283  controlVector->GetControlValue((*processVector)[j]);
284  G4bool activation = processManager->GetProcessActivation(j);
285 
286  if (control != kUnsetControlValue) {
287  if (!TG4Globals::Compare(activation, control)) {
288 
289  // set new process activation
290  G4bool activate;
291  if (control == kInActivate)
292  activate = false;
293  else
294  activate = true;
295 
296  SetProcessActivation(processManager, j, activate);
297  }
298  }
299  }
300  }

357 {
376 
377  if (VerboseLevel() > 1) {
378  G4cout << "TG4PhysicsManager::Gstpar " << param << " " << parval << G4endl;
379  }
380 
381  G4String name = TG4GeometryServices::Instance()->CutName(param);
382  TG4G3Cut cut;
383  if (fG3PhysicsManager->CheckCutWithTheVector(name, parval, cut)) {
384  GstparCut(itmed, cut, parval);
385  fG3PhysicsManager->Lock();
386  }
387  else {
388  TG4G3Control control;
389  TG4G3ControlValue controlValue;

391  name, parval, control, controlValue)) {
392  GstparControl(itmed, control, controlValue);

394  }
395  else if (cut == kNoG3Cuts && control == kNoG3Controls &&
396  fNotImplParNames.find(TString(param)) == fNotImplParNames.end()) {
397 
398  if (TString(param) == "STRA") {

401  text += " Setting via Gstpar is ignored.";
402  TG4Globals::Warning("TG4PhysicsManager", "Gstpar", text);

405  TG4Globals::Warning("TG4PhysicsManager", "Gstpar",
