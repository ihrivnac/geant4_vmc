63 void TG4PrimaryGeneratorAction::TransformPrimaries(G4Event* event)
64 {
67 
68  // Clear the priamary particles info in track manager
69  TG4TrackManager* trackManager = TG4TrackManager::Instance();
70  trackManager->ResetPrimaryParticleIds();
71 
72  // Cache pointers to thread-local objects
73  TVirtualMCStack* mcStack = gMC->GetStack();
74  TG4ParticlesManager* particlesManager = TG4ParticlesManager::Instance();
75 
76  G4int nofParticles = mcStack->GetNtrack();
77  if (nofParticles <= 0) {
78  TG4Globals::Warning("TG4PrimaryGeneratorAction", "TransformPrimaries",
79  "No primary particles found on the stack.");
80  return;
81  }
82 
83  if (VerboseLevel() > 1)
84  G4cout << "TG4PrimaryGeneratorAction::TransformPrimaries: " << nofParticles
85  << " particles" << G4endl;
86 
87  G4PrimaryVertex* previousVertex = 0;
88  G4ThreeVector previousPosition = G4ThreeVector();
89  G4double previousTime = 0.;
90 
91  for (G4int i = 0; i < nofParticles; i++) {
92 
93  // get the particle from the stack
94  TParticle* particle = mcStack->PopPrimaryForTracking(i);
95 
96  if (particle) {
97  // only particles that didn't die (decay) in primary generator
98  // will be transformed to G4 objects
99 
100  // Pass this particle Id (in the VMC stack) to Track manager
101  trackManager->AddPrimaryParticleId(i);
102 
103  // Get particle definition from TG4ParticlesManager
104  //
105  G4ParticleDefinition* particleDefinition =
106  particlesManager->GetParticleDefinition(particle, false);
107 
108  if (!particleDefinition) {
109  TString text = "G4ParticleTable::FindParticle() failed for ";
110  text += TString(particle->GetName());
111  text += " pdgEncoding=";
112  text += particle->GetPdgCode();
113  text += ".";
114  if (fSkipUnknownParticles) {
115  TG4Globals::Warning(
116  "TG4PrimaryGeneratorAction", "TransformPrimaries", text);
117  continue;
118  }
119  else {
120  TG4Globals::Exception(
121  "TG4PrimaryGeneratorAction", "TransformPrimaries", text);
122  }
123  }
124 
125  // Get/Create vertex
126  G4ThreeVector position = particlesManager->GetParticlePosition(particle);
127  G4double time = particle->T() * TG4G3Units::Time();
128  G4PrimaryVertex* vertex;
129  if (i == 0 || previousVertex == 0 || position != previousPosition ||
130  time != previousTime) {
131  // Create a new vertex
132  // in case position and time of gun particle are different from
133  // previous values
134  // (vertex objects are destroyed in G4EventManager::ProcessOneEvent()
135  // when event is deleted)
136  vertex = new G4PrimaryVertex(position, time);
137  event->AddPrimaryVertex(vertex);
138 
139  previousVertex = vertex;
140  previousPosition = position;
141  previousTime = time;
142  }
143  else
144  vertex = previousVertex;
145 
146  // Create a primary particle and add it to the vertex
147  // (primaryParticle objects are destroyed in
148  // G4EventManager::ProcessOneEvent() when event and then vertex is
149  // deleted)
150  G4ThreeVector momentum = particlesManager->GetParticleMomentum(particle);
151  G4double energy = particle->Energy() * TG4G3Units::Energy();
152  G4PrimaryParticle* primaryParticle = new G4PrimaryParticle(
153  particleDefinition, momentum.x(), momentum.y(), momentum.z(), energy);
154 
155  // Set charge
156  G4double charge = particleDefinition->GetPDGCharge();
157  if (G4IonTable::IsIon(particleDefinition) &&
158  particleDefinition->GetParticleName() != "proton") {
159  // Get dynamic charge defined by user
160  TG4UserIon* userIon =
161  particlesManager->GetUserIon(particle->GetName(), false);
162  if (userIon) charge = userIon->GetQ() * eplus;
163  }
164  primaryParticle->SetCharge(charge);
165 
166  // Set polarization
167  TVector3 polarization;
168  particle->GetPolarisation(polarization);
169  primaryParticle->SetPolarization(
170  polarization.X(), polarization.Y(), polarization.Z());
171 
172  // Set weight
173  G4double weight = particle->GetWeight();
174  primaryParticle->SetWeight(weight);
175 
176  // Add primary particle to the vertex
177  vertex->SetPrimary(primaryParticle);
178 
179  // Verbose
180  if (VerboseLevel() > 1) {
181  G4cout << i << "th primary particle: " << G4endl;
182  primaryParticle->Print();
183  }
184  }
185  }

193 void TG4PrimaryGeneratorAction::GeneratePrimaries(G4Event* event)
194 {
196 
197  // Cache pointer to thread-local MC application
198  TVirtualMCApplication* mcApplication = TVirtualMCApplication::Instance();
199 
200  // Begin of event
201  TG4StateManager::Instance()->SetNewState(kInEvent);
202  mcApplication->BeginEvent();
203 
204  // Update cached pointer to MC stack which is set to MC in some application
