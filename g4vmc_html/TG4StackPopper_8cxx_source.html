82 G4VParticleChange* TG4StackPopper::PostStepDoIt(
83  const G4Track& track, const G4Step& /*step*/)
84 {
86 
87  aParticleChange.Initialize(track);
88 
89  if (fMCStack->GetNtrack() == fNofDoneTracks) return &aParticleChange;
90 
91  Int_t currentTrackId = fMCStack->GetCurrentTrackNumber();
92  Int_t nofTracksToPop = fMCStack->GetNtrack() - fNofDoneTracks;
93  aParticleChange.SetNumberOfSecondaries(
94  aParticleChange.GetNumberOfSecondaries() + nofTracksToPop);
95 
96  for (G4int i = 0; i < nofTracksToPop; ++i) {
97 
98  // Pop particle from the stack
99  G4int itrack;
100  TParticle* particle = fMCStack->PopNextTrack(itrack);
101  ++fNofDoneTracks;
102 
103  if (!particle) {
104  TG4Globals::Exception(
105  "TG4StackPopper", "PostStepDoIt", "No particle popped from stack!");
106  return &aParticleChange;
107  }
108 
109  // G4cout << "TG4StackPopper::PostStepDoIt: Popped particle = "
110  // << particle->GetName()
111  // << " trackID = "<< itrack << G4endl;
112 
113  // Create dynamic particle
114  G4DynamicParticle* dynamicParticle =
115  TG4ParticlesManager::Instance()->CreateDynamicParticle(particle);
116  if (!dynamicParticle) {
117  TG4Globals::Exception("TG4StackPopper", "PostStepDoIt",
118  "Conversion from Root particle -> G4 particle failed.");
119  }
120 
121  // Define track
122 
123  G4ThreeVector position =
124  TG4ParticlesManager::Instance()->GetParticlePosition(particle);
125  G4double time = particle->T() * TG4G3Units::Time();
126 
127  G4Track* secondaryTrack = new G4Track(dynamicParticle, time, position);
128 
129  // set track information here to avoid saving track in the stack
130  // for the second time
131  TG4TrackInformation* trackInformation = new TG4TrackInformation(itrack);
132  // the track information is deleted together with its
133  // G4Track object
134  trackInformation->SetIsUserTrack(true);
135  trackInformation->SetPDGEncoding(particle->GetPdgCode());
136  secondaryTrack->SetUserInformation(trackInformation);
137 
138  // Add track as a secondary
139  aParticleChange.AddSecondary(secondaryTrack);
140  }
141 
142  // Set back current track number in the track
143  // (as stack may have changed it with popping particles)
144  fMCStack->SetCurrentTrack(currentTrackId);
145 
146  // Set the kept track status if in exclusive step
147  if (fDoExclusiveStep) {
148  aParticleChange.ProposeTrackStatus(fTrackStatus);
149  fDoExclusiveStep = false;
150  }
151 
152  return &aParticleChange;
153 }
