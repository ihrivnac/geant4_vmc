77 void TG4EventAction::BeginOfEventAction(const G4Event* event)
78 {
80 
81  // reset the tracks counters
82  fTrackingAction->PrepareNewEvent();
83 
84  // fill primary particles in VMC stack if stack is empty
85  if (fMCStack->GetNtrack() == 0) {
86  if (VerboseLevel() > 0)
87  G4cout << "Filling VMC stack with primaries" << G4endl;
88 
89  for (G4int iv = 0; iv < event->GetNumberOfPrimaryVertex(); iv++) {
90  G4PrimaryVertex* vertex = event->GetPrimaryVertex(iv);
91 
92  for (G4int ip = 0; ip < vertex->GetNumberOfParticle(); ip++) {
93  G4PrimaryParticle* particle = vertex->GetPrimary(ip);
94  fTrackManager->PrimaryToStack(vertex, particle);
95  }
96  }
97  }
98 
99  // save the event random number status per event
100  if (fSaveRandomStatus) {
101  G4UImanager::GetUIpointer()->ApplyCommand("/random/saveThisEvent");
102  if (VerboseLevel() > 0) G4cout << "Saving random status: " << G4endl;
103  CLHEP::HepRandom::showEngineStatus();
104  G4cout << G4endl;
105  }
106 
107  if (VerboseLevel() > 0) {
108  G4cout << ">>> Event " << event->GetEventID() << G4endl;
109  fTimer.Start();
110  }
111 }

146  userSD->EndOfEvent();

153

159
