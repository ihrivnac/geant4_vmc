143 G4bool TG4VisManager::CaseInsensitiveEqual(
144  const G4String string1, const G4String string2)
145 {
147 
148  G4int i;
149  if (string1 == string2) return true;
150  // if (string1.length() != string2.length()) return false;
151  for (i = 0; i < G4int(string1.length()); i++) {
152  G4int diff = abs((G4int)string1(i) - (G4int)string2(i));
153  if (diff && (diff != 32)) return false;
154  }
155  if (string2.length() > string1.length()) {
156  if (string2(i) == '_') return true;
157  return false;
158  }
159  return true;
160 }

444 void TG4VisManager::SetColors()
445 {
447 
448  G4LogicalVolumeStore* pLVStore = G4LogicalVolumeStore::GetInstance();
449  const G4LogicalVolume* pLV = 0;
450  // parse the LV tree and set colours according to material density
451  for (G4int i = 0; i < G4int(pLVStore->size()); i++) {
452  pLV = (*pLVStore)[i];
453  // G4cout << "VOLUME : " << pLV->GetName() << G4endl;
454  const G4Material* pMaterial = pLV->GetMaterial();
455  const G4State kState = pMaterial->GetState();
456  G4double density = (pMaterial->GetDensity()) * cm3 / g;
457  G4String nState = "Undefined";
458  G4int colour = 1; // black by default
459  G4double luminosity = 0.;
460  if (kState == kStateUndefined) {
461  nState = "Undefined";
462  }
463  if (kState == kStateSolid) {
464  nState = "Solid";
465  if (density < 2) {
466  colour = 17; // red
467  luminosity = 25 - 25 * density / 2;
468  }
469  else if (density < 3) {
470  colour = 117; // blue
471  luminosity = 25 - 25 * (density - 2);
472  }
473  else if (density < 10) {
474  colour = 67; // green
475  luminosity = 25 - 25 * (density - 5) / 5;
476  }
477  else if (density < 15) {
478  colour = 92; // cyan
479  luminosity = 25 - 25 * (density - 10) / 5;
480  }
481  else if (density < 20) {
482  colour = 8; // black
483  luminosity = 9 - 9 * (density - 15) / 5;
484  }
485  }
486  if (kState == kStateLiquid) {
487  nState = "Liquid";
488  colour = 142; // violet
489  luminosity = 25 - 25 * density / 2;
490  }
491  if (kState == kStateGas) {
492  nState = "Gas";
493  if (density < 0.001) {
494  colour = 42;
495  } // yellow
496  else if (density < 0.002) {
497  colour = 27;
498  } // light red
499  else if (density < 0.003) {
500  colour = 77;
501  } // light green
502  else {
503  colour = 102;
504  } // light cyan
505  luminosity = 0;
506  }
507  if (luminosity < 0) luminosity = 0;
508  colour += (G4int)luminosity;
509  // Setting the corresponding colour
510  Gsatt(pLV->GetName(), "COLO", colour);
511  }
