60 void TG4SDConstruction::CreateSD(
61  G4LogicalVolume* lv, TVirtualMCSensitiveDetector* userSD) const
62 {
64 
65  TG4GeometryServices* geometryServices = TG4GeometryServices::Instance();
66  G4SDManager* pSDManager = G4SDManager::GetSDMpointer();
67 
68  G4String sdName;
69  if (userSD) {
70  sdName = userSD->GetName();
71  }
72  else {
73  G4String lvName = lv->GetName();
74  sdName = "/" + lvName;
75  // cut copy number from sdName
76  sdName = geometryServices->UserVolumeName(sdName);
77  }
78 
79  // create/retrieve the sensitive detector
80  G4VSensitiveDetector* sd = 0;
81  sd = pSDManager->FindSensitiveDetector(sdName, false);
82  if (!sd) {
83 
84  G4int mediumId = TG4GeometryServices::Instance()->GetMediumId(lv);
85 
86  TG4SensitiveDetector* newSD = 0;
87  if (fIsGflash) {
88  newSD = new TG4GflashSensitiveDetector(sdName, mediumId);
89  }
90  else if (userSD) {
91  newSD = new TG4SensitiveDetector(userSD, mediumId, fExclusiveSDScoring);
92  }
93  else {
94  newSD = new TG4SensitiveDetector(sdName, mediumId);
95  }
96  pSDManager->AddNewDetector(newSD);
97 
98  if (VerboseLevel() > 1) {
99  G4cout << "Sensitive detector " << sdName << " ID=" << newSD->GetID()
100  << " medium ID=" << newSD->GetMediumID() << " has been created."
101  << G4endl;
102  }
103  sd = newSD;
104  }
105  lv->SetSensitiveDetector(sd);
106 }

109 void TG4SDConstruction::FillSDSelectionFromTGeo()
110 {
114 
115  if (!gGeoManager) {
116  TG4Globals::Exception(
117  "TG4SDServices", "FillSDSelectionFromTGeo", "TGeo manager not defined.");
118  }
119 
120  TObjArray* volumes = gGeoManager->GetListOfVolumes();
121  TIterator* it = volumes->MakeIterator();
122  TGeoVolume* volume = 0;
123  while ((volume = (static_cast<TGeoVolume*>(it->Next())))) {
124  if (TString(volume->GetOption()) == TString(fSVLabel.data())) {
125  G4cout << "Adding volume " << volume->GetName() << " in SD selection"
126  << G4endl;
127  fSelection.insert(volume->GetName());
128  }
129  }
130 
131  if (!fSelection.size()) {
132  TString text = "No volumes with the option = \"";
133  text += TString(fSVLabel.data());
134  text += "\" were found in TGeo geometry.\n";
135  text += " The SD selection will not be applied.";
136  TG4Globals::Warning("TG4SDServices", "FillSDSelectionFromTGeo", text);
137  }

205 void TG4SDConstruction::Construct()
206 {
209 
210  if (VerboseLevel() > 1) G4cout << "TG4SDConstruction::Construct" << G4endl;
211 
212  G4bool isMaster = !G4Threading::IsWorkerThread();
213 
214  // Get volumes selected as SD in TGeo geometry
215  if (fSelectionFromTGeo && isMaster) FillSDSelectionFromTGeo();
216 
217  // Construct user SDs
218  TG4StateManager::Instance()->SetNewState(kConstructSD);
219  TVirtualMCApplication::Instance()->ConstructSensitiveDetectors();
220  TG4StateManager::Instance()->SetNewState(kNotInApplication);
221 
222  G4bool isUserSD = false;
223  G4LogicalVolumeStore* lvStore = G4LogicalVolumeStore::GetInstance();
224 
225  for (G4int i = 0; i < G4int(lvStore->size()); i++) {
226  G4LogicalVolume* lv = (*lvStore)[i];
227 
228  // Check if a user SD is defined
229  TVirtualMCSensitiveDetector* userSD =
230  TG4SDServices::Instance()->GetUserSD(lv->GetName(), false);
231 
232  // Create SD calling user sensitive detector
233 
234  if (userSD) {
235  CreateSD(lv, userSD);
236  // if ( isMaster ) TG4SDServices::Instance()->MapVolume(lv, sdID);
237  isUserSD = true;
238  }
239  else {
240  // Create SD calling MCApplication::Stepping
241  // if exclusive scoring via user sensitive detectors is not activated and
242  // if selection is empty or if selection is defined and the volume name is
243  // in selection
244  if ((!fExclusiveSDScoring) &&
245  (!fSelection.size() ||
246  fSelection.find(lv->GetName()) != fSelection.end())) {
247 
248  CreateSD(lv, 0);
249  }
250  }
251  }
252 
253  // Define volume Ids if VMC SD is not defined for all volumes
254  // (either due to user defined SDs or user selection of sensitive volumes)
255  if (isMaster) {
256  if (isUserSD) {
257  MapVolumesToInstanceIds();
258  }
259  else {
260  MapVolumesToSDIds();
261  }
262  }
263 
264  // Initialize user geometry
265  TG4StateManager::Instance()->SetNewState(kInitGeometry);
266  TVirtualMCApplication::Instance()->InitGeometry();
267  TG4StateManager::Instance()->SetNewState(kNotInApplication);
268 
269  // Initialize user SDs
270  if (TG4SDServices::Instance()->GetUserSDs()) {
271  for (auto& userSD : (*TG4SDServices::Instance()->GetUserSDs())) {
272  userSD->Initialize();
273  }
274  }
275 
276  if (VerboseLevel() > 1) {
277  TG4SDServices::Instance()->PrintVolNameToIdMap();
278  TG4SDServices::Instance()->PrintVolIdToLVMap();
279  if (fSelection.size()) {
280  TG4SDServices::Instance()->PrintSensitiveVolumes();
281  }
282  if (TG4SDServices::Instance()->GetUserSDs()) {
283  TG4SDServices::Instance()->PrintUserSensitiveDetectors();
284  }
285  }
286 
287  if (VerboseLevel() > 1)
288  G4cout << "TG4SDConstruction::Construct done" << G4endl;
289 }
