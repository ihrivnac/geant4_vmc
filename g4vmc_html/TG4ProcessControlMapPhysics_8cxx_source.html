229 void TG4ProcessControlMapPhysics::ConstructProcess()
230 {
233 
234  TG4ProcessControlMap* controlMap = TG4ProcessControlMap::Instance();
235  G4bool success = true;
236 
237  auto aParticleIterator = GetParticleIterator();
238  aParticleIterator->reset();
239  while ((*aParticleIterator)()) {
240  // skip iteration if particle does not have a process manager
241  if (!aParticleIterator->value()->GetProcessManager()) continue;
242 
243  G4ProcessVector* processVector =
244  aParticleIterator->value()->GetProcessManager()->GetProcessList();
245 
246  for (G4int i = 0; i < processVector->length(); i++) {
247 
248  G4String processName = (*processVector)[i]->GetProcessName();
249 
250  if (controlMap->GetControl(processName) == kNoG3Controls &&
251  processName != "Transportation" &&
252  processName != "MonopoleTransportation" &&
253  processName != "PositronNuclear" &&
254  processName != "positronNuclear" && processName != "ElectroNuclear" &&
255  processName != "electronNuclear" && processName != "photoNuclear" &&
256  processName != "photonNuclear" && processName != "Scintillation" &&
257  processName != "OpMieHG" && processName != "OpWLS" &&
258  processName != "MinEkineCuts" && processName != "G4MinEkineCuts" &&
259  processName != "MaxTimeCuts" && processName != "stackPopper" &&
260  processName != "StepLimiter" && processName != "GammaXTRadiator" &&
261  processName != "StrawXTRadiator" &&
262  processName != "RegularXTRadiator" &&
263  processName != "G4MaxTimeCuts") {
264 
265  G4String text = "Unknown process control for ";
266  text += processName;
267  TG4Globals::Warning(
268  "TG4ProcessControlMapPhysics", "ConstructProcess", text);
269  success = false;
270  }
271  }
272  }
273 
274  if (VerboseLevel() > 0) {
275  if (success)
276  G4cout << "### Processes mapped to VMC controls ok." << G4endl;
277  else
278  G4cout << "### Processes mapped to VMC controls: some proceess unknown."
279  << G4endl;
