50 {
53 
54  // set condition
55  *condition = NotForced;
56  G4double proposedStep = DBL_MAX;
57 
58  // get limits
59 #ifdef MCDEBUG
60  TG4Limits* limits = TG4GeometryServices::Instance()->GetLimits(
61  track.GetVolume()->GetLogicalVolume()->GetUserLimits());
62 #else
63  TG4Limits* limits =
64  (TG4Limits*)track.GetVolume()->GetLogicalVolume()->GetUserLimits();
65 #endif
66 
67  if (!limits) return proposedStep;
68 
69  // tracks flagged to stop
70  TG4TrackInformation* trackInformation =
71  fTrackManager->GetTrackInformation(&track);
72  if (trackInformation && trackInformation->IsStop()) {
73  return 0.;
74  }
75 
76  // min kinetic energy (from limits)
77  G4double minEkine = GetMinEkine(*limits, track);
78  if (track.GetKineticEnergy() <= minEkine) return 0.;
79 
80  // max track length
81  proposedStep =
82  (limits->GetUserMaxTrackLength(track) - track.GetTrackLength());
83  if (proposedStep < 0.) return 0.;
84 
85  // max time limit
86  G4double tlimit = limits->GetUserMaxTime(track);
87  if (tlimit < DBL_MAX) {
88  G4double beta = (track.GetDynamicParticle()->GetTotalMomentum()) /
89  (track.GetTotalEnergy());
90  G4double dTime = (tlimit - track.GetGlobalTime());
91  G4double temp = beta * c_light * dTime;
92  if (temp < 0.) {
93  return 0.;
94  }
95  if (proposedStep > temp) {
96  proposedStep = temp;
97  }
98  }
99 
100  // min remaining range
101  // (only for charged particle except for chargedGeantino)
102  G4double rmin = limits->GetUserMinRange(track);
103  if (rmin > DBL_MIN) {
104  G4ParticleDefinition* particle = track.GetDefinition();
105  if ((particle->GetPDGCharge() != 0.) && (particle->GetPDGMass() > 0.0)) {
106  G4double ekin = track.GetKineticEnergy();
107  const G4MaterialCutsCouple* couple = track.GetMaterialCutsCouple();
108  G4double rangeNow = fLossTableManager->GetRange(particle, ekin, couple);
109  G4double temp = rangeNow - rmin;
110  if (temp < 0.) {
111  return 0.;
112  }
113  if (proposedStep > temp) {
114  proposedStep = temp;
115  }
116  }
117  }
118  return proposedStep;
119 }
