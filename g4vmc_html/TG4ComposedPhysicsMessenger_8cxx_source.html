225 void TG4ComposedPhysicsMessenger::SetNewValue(
226  G4UIcommand* command, G4String newValue)
227 {
229 
230  if (command == fRangeGammaCutCmd) {
231  G4double cut = fRangeGammaCutCmd->GetNewDoubleValue(newValue);
232  TG4PhysicsManager::Instance()->SetCutForGamma(cut);
233  fPhysicsList->SetCutForGamma(cut);
234  }
235  else if (command == fRangeElectronCutCmd) {
236  G4double cut = fRangeElectronCutCmd->GetNewDoubleValue(newValue);
237  TG4PhysicsManager::Instance()->SetCutForElectron(cut);
238  fPhysicsList->SetCutForElectron(cut);
239  }
240  else if (command == fRangePositronCutCmd) {
241  G4double cut = fRangePositronCutCmd->GetNewDoubleValue(newValue);
242  TG4PhysicsManager::Instance()->SetCutForPositron(cut);
243  fPhysicsList->SetCutForPositron(cut);
244  }
245  else if (command == fRangeProtonCutCmd) {
246  G4double cut = fRangeProtonCutCmd->GetNewDoubleValue(newValue);
247  TG4PhysicsManager::Instance()->SetCutForProton(cut);
248  fPhysicsList->SetCutForProton(cut);
249  }
250  else if (command == fRangeAllCutCmd) {
251  G4double cut = fRangeAllCutCmd->GetNewDoubleValue(newValue);
252  TG4PhysicsManager::Instance()->SetCutForGamma(cut);
253  TG4PhysicsManager::Instance()->SetCutForElectron(cut);
254  TG4PhysicsManager::Instance()->SetCutForPositron(cut);
255  TG4PhysicsManager::Instance()->SetCutForProton(cut);
256  fPhysicsList->SetCutForGamma(cut);
257  fPhysicsList->SetCutForElectron(cut);
258  fPhysicsList->SetCutForPositron(cut);
259  fPhysicsList->SetCutForProton(cut);
260  }
261  else if (command == fProductionCutsTableEnergyRangeCmd) {
262  // tokenize parameters in a vector
263  std::vector<G4String> parameters;
264  G4Analysis::Tokenize(newValue, parameters);
265 
266  G4int counter = 0;
267  G4double minEnergy = G4UIcommand::ConvertToDouble(parameters[counter++]);
268  G4double minEUnit = G4UnitDefinition::GetValueOf(parameters[counter++]);
269  G4double maxEnergy = G4UIcommand::ConvertToDouble(parameters[counter++]);
270  G4double maxEUnit = G4UnitDefinition::GetValueOf(parameters[counter++]);
271  fPhysicsList->SetProductionCutsTableEnergyRange(
272  minEnergy * minEUnit, maxEnergy * maxEUnit);
273  }
274  else if (command == fSetGammaToMuonsCrossSectionFactorCmd) {
275  G4double value = G4UIcommand::ConvertToDouble(newValue);
276  fPhysicsList->SetGammaToMuonsCrossSectionFactor(value);
277  }
278  else if (command == fPrintAllProcessesCmd) {
279  fPhysicsList->PrintAllProcesses();
280  }
281  else if (command == fDumpAllProcessesCmd) {
282  fPhysicsList->DumpAllProcesses();
283  }
284  else if (command == fPrintProcessMCMapCmd) {
285  TG4ProcessMCMap::Instance()->PrintAll();
286  }
287  else if (command == fPrintProcessControlMapCmd) {
288  TG4ProcessControlMap::Instance()->PrintAll();
289  }
290  else if (command == fPrintVolumeLimitsCmd) {
291  TG4GeometryServices::Instance()->PrintVolumeLimits(newValue);
292  }
293  else if (command == fPrintGlobalCutsCmd) {
294  TG4G3PhysicsManager::Instance()->GetCutVector()->Print();
295  }
296  else if (command == fPrintGlobalControlsCmd) {
297  TG4G3PhysicsManager::Instance()->GetControlVector()->Print();
298  }
299  else if (command == fG4NeutronHPVerboseCmd) {
300  G4NeutronHPManager::GetInstance()->SetVerboseLevel(
301  fG4NeutronHPVerboseCmd->GetNewIntValue(newValue));
302  }
303  else if (command == fG4HadronicProcessStoreVerboseCmd) {
304  G4HadronicProcessStore::Instance()->SetVerbose(
305  fG4HadronicProcessStoreVerboseCmd->GetNewIntValue(newValue));
306  }
