95 void TG4ParticlesManager::AddParticleToPdgDatabase(
96  const G4String& name, G4ParticleDefinition* particleDefinition)
97 {
99 
100  // Return if particle was already added
101  G4int pdgEncoding = particleDefinition->GetPDGEncoding();
102  TParticlePDG* particlePDG =
103  TDatabasePDG::Instance()->GetParticle(pdgEncoding);
104  if (particlePDG) return;
105 
106  // Get particle data
107  G4String g4Name = particleDefinition->GetParticleName();
108  G4int pdgQ = G4int(particleDefinition->GetPDGCharge() / eplus);
109  // !! here we do not save dynamic charge but the static one
110  G4String g4Type = particleDefinition->GetParticleType();
111  G4String rootType = g4Type;
112  if (g4Type == "nucleus" || g4Type == "anti_nucleus") rootType = "Ion";
113 
114  if (VerboseLevel() > 1) {
115  G4cout << "Adding particle to TDatabasePDG " << G4endl;
116  G4cout << " name: " << g4Name << G4endl;
117  G4cout << " g4name: " << name << G4endl;
118  G4cout << " PDG: " << pdgEncoding << G4endl;
119  G4cout << " pdgQ: " << pdgQ << G4endl;
120  G4cout << " type: " << rootType << G4endl;
121  }
122 
123  // Add particle to TDatabasePDG
124 #ifdef G4MULTITHREADED
125  G4AutoLock lm(&addParticleMutex);
126 #endif
127  TDatabasePDG::Instance()->AddParticle(name, g4Name,
128  particleDefinition->GetPDGMass() / TG4G3Units::Energy(),
129  particleDefinition->GetPDGStable(),
130  particleDefinition->GetPDGWidth() / TG4G3Units::Energy(), pdgQ * 3,
131  rootType, pdgEncoding);
132 #ifdef G4MULTITHREADED
133  lm.unlock();
134 #endif
135 }

142 void TG4ParticlesManager::DefineParticles()
143 {
146 
147  const Int_t kspe = 50000000;
148  TDatabasePDG* pdgDB = TDatabasePDG::Instance();
149 
150  // optical phothon
151  if (!pdgDB->GetParticle(kspe + 50))
152  pdgDB->AddParticle(
153  "Cherenkov", "Cherenkov", 0, kFALSE, 0, 0, "Special", kspe + 50);
154  fParticleNameMap.Add("opticalphoton", "Cherenkov");
155 
156  // feedback phothon
157  if (!pdgDB->GetParticle(kspe + 51))
158  pdgDB->AddParticle("FeedbackPhoton", "FeedbackPhoton", 0, kFALSE, 0, 0,
159  "Special", kspe + 51);
160  fParticleNameMap.AddInverse("opticalphoton", "FeedbackPhoton");
161 
162  // generic ion
163  // This particle should not appear in tracking (as it is commented
164  // in class G4GenericIon), but as it does, we map it anyway
165  if (!pdgDB->GetParticle(kspe + 60))
166  pdgDB->AddParticle(
167  "GenericIon", "GenericIon", 0.938272, kTRUE, 0, 1, "Special", kspe + 60);
168  fParticleNameMap.Add("GenericIon", "GenericIon");
169 
170  // Light ions
171  // Get PDG codes from Geant4 the rest as in TGeant3
172  const Double_t kGeV = 0.9314943228;
173  const Double_t kHslash = 1.0545726663e-27;
174  const Double_t kErgGeV = 1 / 1.6021773349e-3;
175  const Double_t kHshGeV = kHslash * kErgGeV;
176  const Double_t kYearsToSec = 3600 * 24 * 365.25;
177 
178  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
179  G4ParticleDefinition* particle;
180 
181  particle = particleTable->FindParticle("deuteron");
182  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
183  pdgDB->AddParticle("Deuteron", "Deuteron", 2 * kGeV + 8.071e-3, kTRUE, 0, 3,
184  "Ion", particle->GetPDGEncoding());
185  }
186 
187  particle = particleTable->FindParticle("triton");
188  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
189  pdgDB->AddParticle("Triton", "Triton", 3 * kGeV + 14.931e-3, kFALSE,
190  kHshGeV / (12.33 * kYearsToSec), 3, "Ion", particle->GetPDGEncoding());
191  }
192 
193  particle = particleTable->FindParticle("alpha");
194  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
195  pdgDB->AddParticle("Alpha", "Alpha", 4 * kGeV + 2.424e-3, kTRUE,
196  kHshGeV / (12.33 * kYearsToSec), 6, "Ion", particle->GetPDGEncoding());
197  }
198 
199  particle = particleTable->FindParticle("He3");
200  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
201  pdgDB->AddParticle("HE3", "HE3", 3 * kGeV + 14.931e-3, kFALSE, 0, 6, "Ion",
202  particle->GetPDGEncoding());
203  }
204 
205  // Light anti-ions
206  // Get parameters from Geant4
207 
208  particle = particleTable->FindParticle("anti_deuteron");
209  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
210  pdgDB->AddParticle("AntiDeuteron", "AntiDeuteron", 1.875613, kTRUE, 0, -3,
211  "Ion", particle->GetPDGEncoding());
212  }
213 
214  particle = particleTable->FindParticle("anti_triton");
215  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
216  pdgDB->AddParticle("AntiTriton", "AntiTriton", 2.808921, kTRUE, 0, -3,
217  "Ion", particle->GetPDGEncoding());
218  }
219 
220  particle = particleTable->FindParticle("anti_alpha");
221  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
222  pdgDB->AddParticle("AntiAlpha", "AntiAlpha", 3.727379, kTRUE, 0, -6, "Ion",
223  particle->GetPDGEncoding());
224  }
225 
226  particle = particleTable->FindParticle("anti_He3");
227  if (particle && !pdgDB->GetParticle(particle->GetPDGEncoding())) {
228  pdgDB->AddParticle("AntiHE3", "AntiHE3", 2.808391, kTRUE, 0, -6, "Ion",
229  particle->GetPDGEncoding());
230  }
231 
232  // monopole
233  particle = particleTable->FindParticle("monopole");
234  if (particle) {
235  if (!pdgDB->GetParticle(60000000)) {
236  G4cout << "Adding monnopole in TDatabase with mass "
237  << particle->GetPDGMass() / GeV << G4endl;
238  pdgDB->AddParticle("Monopole", "Monopole", particle->GetPDGMass() / GeV,
239  kTRUE, particle->GetPDGCharge() / eplus * 3.,
240  static_cast<G4Monopole*>(particle)->MagneticCharge() / eplus * 3.,
241  "Special", 60000000);
242  fParticleNameMap.Add("monopole", "monopole");
243  }
244  else {
245  TG4Globals::Warning("TG4ParticlesManager", "DefineParticles",
246  "Cannot add monopole with PDG=60000000 in TDatabasePDG.");
247  }
248  }
249 
250  // geantino
251  fParticleNameMap.Add("geantino", "Rootino");
252  fParticleNameMap.Add("chargedgeantino", "ChargedRootino");
253  // ChargedRootino does not exist in Root particle database
254  // but the user can just change the title of Rootino to ChargedRootino
255  // to get it interpreted as chargedgeantino:
256  // TParticlePDG* rootino = pdgTable->GetParticle("Rootino");
257  // if (rootino) rootino->SetTitle("ChargedRootino");
258 
259  if (VerboseLevel() > 1) {
260  fParticleNameMap.PrintAll();
261  }
262 }

323  const G4String& name, G4int Z, G4int A, G4int Q, G4double excEnergy)
324 {
326 
327  if (VerboseLevel() > 1) {
328  G4cout << "Adding ion with Z, A, Q, excEnergy [keV]: " << Z << " " << A
329  << " " << Q << " " << excEnergy / keV << G4endl;
330  }
331 
332  // Get G4 ion particle definition
333  // (Ion is created if it does not yet exist)
334 #if G4VERSION_NUMBER >= 1000
335  G4ParticleDefinition* particleDefinition =

339  G4ParticleTable::GetParticleTable()->GetIon(Z, A, excEnergy);
340 #endif
341 
342  if (!particleDefinition) {
343  TString text = "Z, A, excEnergy [keV]: ";
344  text += Z;
345  text += " ";
346  text += A;
347  text += " ";
348  text += excEnergy / keV;
349  text += ".";
350  TG4Globals::Exception("TG4ParticlesManager", "AddIon",
351  text + TG4Globals::Endl() +
352 #if G4VERSION_NUMBER >= 1000
353  "G4IonTable::GetIon() failed.");
354 #else
355  "G4ParticleTable::GetIon() failed.");
356 #endif
357  }
358 
359  // Add ion to TDatabasePDG
360  AddParticleToPdgDatabase(name, particleDefinition);
361 
362  // Add ion to the map to be able to retrieve later its charge
363  fUserIonMap[name] =
364  new TG4UserIon(name, particleDefinition->GetPDGEncoding(), Q);

368 Bool_t TG4ParticlesManager::SetDecayMode(
369  Int_t pdg, Float_t bratio[6], Int_t mode[6][3])
370 {
372 
373  // Check input pdg
374  if (pdg == 0) {
375  TG4Globals::Exception("TG4ParticlesManager", "SetDecayMode",
376  "Cannot define decay mode for particle with PDG=0");
377  }
378 
379  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
380  G4ParticleDefinition* particleDefinition = particleTable->FindParticle(pdg);
381 
382  if (!particleDefinition) {
383  TString pdgs;
384  pdgs += pdg;
385  TG4Globals::Exception("TG4ParticlesManager", "SetDecayMode",
386  "Particle PDG=" + pdgs + " was not found in G4ParticleTable.");
387  }
388 
389  G4String particleName = particleDefinition->GetParticleName();
390  G4DecayTable* decayTable = new G4DecayTable();
391 
392  // Loop over decay channels
393  //
394  for (Int_t kz = 0; kz < 6; ++kz) {
395 
396  // Do not fill empty channels
397  if (bratio[kz] == 0.) break;
398 
399  // Fill names of daughters
400  G4int nofDaughters = 0;
401  std::vector<G4String> daughtersNames(3);
402  for (G4int i = 0; i < 3; i++) {
403  daughtersNames[i] = "";
404  if (mode[kz][i] != 0) {
405  G4ParticleDefinition* daughter =
406  particleTable->FindParticle(mode[kz][i]);
407  if (!daughter) {
408  TString pdgs;
409  pdgs += mode[kz][i];
410  TG4Globals::Exception("TG4ParticlesManager", "SetDecayMode",
411  "Daughter particle PDG=" + pdgs +
412  " was not found in G4ParticleTable.");
413  }
414  daughtersNames[i] = daughter->GetParticleName();
415  ++nofDaughters;
416  }
417  }
418 
419  // Define decay channel
420  G4VDecayChannel* decayChannel =
421  new G4PhaseSpaceDecayChannel(particleName, bratio[kz] / 100.0,
422  nofDaughters, daughtersNames[0], daughtersNames[1], daughtersNames[2]);
423  decayTable->Insert(decayChannel);
424  }
425 
426  particleDefinition->SetDecayTable(decayTable);
427 
428  if (VerboseLevel() > 1) particleDefinition->DumpTable();
429 
430  return true;
431 }

462 {
466 
467  // Get PDG encoding from G4 particle definition
468  G4int pdgEncoding = particle->GetPDGEncoding();
469  if (pdgEncoding) {
470  // Add particle to TDatabasePDG
471  if (!TDatabasePDG::Instance()->GetParticle(pdgEncoding))
472  AddParticleToPdgDatabase(particle->GetParticleName(), particle);
473  return pdgEncoding;
474  }
475 
476  // Get PDG encoding from TDatabasePDG if not defined in Geant4
477 
478  // get particle name from the name map

480  G4String tname = fParticleNameMap.GetSecond(g4name);
481  if (tname == "ChargedRootino") tname = "Rootino";
482  // special treatment for Rootino
483  // user can reset the particle title to ChargedRootino to interpret
484  // Rootino as chargedgeantino
485 
486  if (tname == "Undefined") {
487  particle->DumpTable();
488  TG4Globals::Exception("TG4ParticlesManager", "GetPDGEncoding",
489  "Particle " + TString(g4name) + " was not found in the name map.");
490  }
491 
492  // get particle from TDatabasePDG
493  TDatabasePDG* pdgDB = TDatabasePDG::Instance();
494  TParticlePDG* tparticle = pdgDB->GetParticle(tname);
495  if (!tparticle) {
496  TG4Globals::Exception("TG4ParticlesManager", "GetPDGEncoding",
497  "Particle " + TString(tname) + " was not found in TDatabasePDG.");
