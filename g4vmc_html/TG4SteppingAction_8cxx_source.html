76 void TG4SteppingAction::ProcessTrackIfLooping(const G4Step* step)
77 {
79 
80  G4Track* track = step->GetTrack();
81  G4int stepNumber = track->GetCurrentStepNumber();
82 
83  if (fLoopStepCounter) {
84  if (stepNumber == 1) {
85  // reset parameters at beginning of tracking
86  fpSteppingManager->SetVerboseLevel(fStandardVerboseLevel);
87  fLoopStepCounter = 0;
88  // necessary in case particle has reached fMaxNofSteps
89  // but has stopped before processing kMaxNofLoopSteps
90  }
91  else {
92  // count steps after detecting looping track
93  fLoopStepCounter++;
94  if (fLoopStepCounter == kMaxNofLoopSteps) {
95 
96  // stop the looping track
97  track->SetTrackStatus(fStopAndKill);
98 
99  // reset parameters back
100  fpSteppingManager->SetVerboseLevel(fStandardVerboseLevel);
101  fLoopStepCounter = 0;
102  }
103  }
104  }
105  else if (stepNumber > fMaxNofSteps) {
106 
107  // print looping info
108  if (fLoopVerboseLevel > 0) {
109  G4cout << "*** Particle reached max step number (" << fMaxNofSteps
110  << "). ***" << G4endl;
111  if (fStandardVerboseLevel == 0) PrintTrackInfo(track);
112  }
113 
114  // keep standard verbose level
115  if (fStandardVerboseLevel < 0)
116  fStandardVerboseLevel = fpSteppingManager->GetverboseLevel();
117 
118  // set loop verbose level
119  fpSteppingManager->SetVerboseLevel(fLoopVerboseLevel);
120 
121  // start looping counter
122  fLoopStepCounter++;
123  }
124 }

127 void TG4SteppingAction::ProcessTrackIfOutOfRegion(const G4Step* step)
128 {
130 
131  G4ThreeVector position = step->GetPostStepPoint()->GetPosition();
132  position /= TG4G3Units::Length();
133 
134  if (position.perp() > fMCApplication->TrackingRmax() ||
135  std::abs(position.z()) > fMCApplication->TrackingZmax()) {
136 
137  // print looping info
138  if (fLoopVerboseLevel > 0) {
139  G4cout << "*** Particle has reached user defined tracking region. ***"
140  << G4endl;
141  if (fStandardVerboseLevel == 0) PrintTrackInfo(step->GetTrack());
142  }
143 
144  // stop the track
145  step->GetTrack()->SetTrackStatus(fStopAndKill);
146  }
147 }

150 void TG4SteppingAction::ProcessTrackIfBelowCut(const G4Step* step)
151 {
153 
154  if (step->GetSecondary()->size() == 2 &&
155  ((*step->GetSecondary())[0]->GetCreatorProcess()->GetProcessName() ==
156  "muPairProd")) {
157 
158  G4double minEtotPair =
159  fStepManager->GetCurrentLimits()->GetCutVector()->GetMinEtotPair();
160 
161  // G4cout << "minEtotPair[GeV]= " << minEtotPair << G4endl;
162 
163  if (minEtotPair > 0. && (*step->GetSecondary())[0]->GetTotalEnergy() +
164  (*step->GetSecondary())[1]->GetTotalEnergy() <
165  minEtotPair) {
166  // G4cout << "In stepping action: going to flag pair to stop" << G4endl;
167  fTrackManager->SetParentToTrackInformation(step->GetTrack());
168  fTrackManager->GetTrackInformation((*step->GetSecondary())[0])
169  ->SetStop(true);
170  fTrackManager->GetTrackInformation((*step->GetSecondary())[1])
171  ->SetStop(true);
172  }

177 void TG4SteppingAction::ProcessTrackOnBoundary(const G4Step* step)
178 {
180 
181  // let sensitive detector process boundary step
182  // if crossing geometry border
183  // (this ensures compatibility with G3 that
184  // makes boundary step of zero length)
185  if (step->GetTrack()->GetNextVolume() != 0) {
186 
187  // set back max step limit if it has been modified on fly by user
188  G4UserLimits* modifiedLimits = fStepManager->GetLimitsModifiedOnFly();
189 
190  if (modifiedLimits) {
191  G4UserLimits* nextLimits = step->GetPostStepPoint()
192  ->GetPhysicalVolume()
193  ->GetLogicalVolume()
194  ->GetUserLimits();
195 
196  if (nextLimits != modifiedLimits) fStepManager->SetMaxStepBack();
197  }
198 
199  if (step->GetTrack()->GetTrackStatus() == fAlive) {
200 #ifdef MCDEBUG
201  TG4SensitiveDetector* tsd =
202  TG4SDServices::Instance()->GetSensitiveDetector(
203  step->GetPostStepPoint()
204  ->GetPhysicalVolume()
205  ->GetLogicalVolume()
206  ->GetSensitiveDetector());
207 
208  if (tsd) tsd->ProcessHitsOnBoundary((G4Step*)step);
209 #else
210  TG4SensitiveDetector* tsd =
211  (TG4SensitiveDetector*)step->GetPostStepPoint()
212  ->GetPhysicalVolume()
213  ->GetLogicalVolume()
214  ->GetSensitiveDetector();
215 
216  if (tsd) tsd->ProcessHitsOnBoundary((G4Step*)step);
217 #endif
218  }
219  }
