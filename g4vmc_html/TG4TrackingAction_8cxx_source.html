161 void TG4TrackingAction::PreUserTrackingAction(const G4Track* track)
162 {
164 
165  // do not call this function more than once
166  if (track->GetTrackID() == fCurrentTrackID) return;
167 
168  // keep this track number for the check above
169  fCurrentTrackID = track->GetTrackID();
170 
171  G4bool isFirstStep = (track->GetCurrentStepNumber() == 0);
172 
173  // finish previous primary track first
174  if (track->GetParentID() == 0 && isFirstStep) {
175  FinishPrimaryTrack();
176  }
177 
178  // initialize special controls manager
179  if (fSpecialControls) fSpecialControls->StartTrack(track);
180 
181  // reset stack popper (if activated
182  if (fStackPopper) fStackPopper->Reset();
183 
184  // set step manager status
185  if (isFirstStep) {
186  fStepManager->SetStep((G4Track*)track, kVertex);
187  }
188 
189  // set track information
190  G4int trackId =
191  fTrackManager->SetTrackInformation(track, fOverwriteLastTrack);
192  fMCStack->SetCurrentTrack(trackId);
193 
194  if (isFirstStep) {
195  if (track->GetParentID() == 0) {
196  fPrimaryTrackID = track->GetTrackID();
197 
198  // begin this primary track
199  fMCApplication->BeginPrimary();
200 
201  // set saving flag
202  fTrackSaveControl = kDoNotSave;
203  }
204  else {
205  // set saving flag
206  fTrackSaveControl = (fTrackManager->IsUserTrack(track))
207  ? kDoNotSave
208  : fTrackManager->GetTrackSaveControl();
209  }
210 
211  // save track in stack
212  if (fTrackSaveControl == kSaveInPreTrack) {
213  fTrackManager->TrackToStack(track, fOverwriteLastTrack);
214 
215  // Notify a stack popper (if activated) about saving this secondary
216  if (fStackPopper) fStackPopper->Notify();
217  }
218  }
219 
220  // verbose
221  if (track->GetTrackID() == fNewVerboseTrackID) {
222  fpTrackingManager->SetVerboseLevel(fNewVerboseLevel);
223  }
224 
225  // VMC application pre track action
226  if (isFirstStep) {
227  fMCApplication->PreTrack();
228 
229  // call pre-tracking action of derived class
230  PreTrackingAction(track);
231 
232  if (track->GetTrackStatus() != fStopAndKill) {
233  // Let sensitive detector process vertex step
234  UserProcessHits(track);
235  }
236  }
237 }

240 void TG4TrackingAction::PostUserTrackingAction(const G4Track* track)
241 {
243 
244 #ifdef STACK_WITH_KEEP_FLAG
245  // Remember whether this track should be kept in the stack
246  // or can be overwritten:
247  // the track will not be overwritten if it was flagged in the stack
248  // to be kept or if it has produced any secondary particles.
249  fOverwriteLastTrack = (!fMCStack->GetKeepCurrentTrack()) &&
250  (!fpTrackingManager->GimmeSecondaries() ||
251  fpTrackingManager->GimmeSecondaries()->size() == 0);
252  // Experimental code with flagging tracks in stack for overwrite;
253  // not yet available in distribution
254 #else
255  fOverwriteLastTrack = false;
256 #endif
257 
258  // restore processes activation
259  if (fSpecialControls && fSpecialControls->IsApplicable())
260  fSpecialControls->RestoreProcessActivations();
261 
262  // set back max step limit if it has been modified on fly by user
263  if (fStepManager->GetLimitsModifiedOnFly()) fStepManager->SetMaxStepBack();
264 
265  // set parent track particle index to the secondary tracks
266  fTrackManager->SetParentToTrackInformation(track);
267 
268  // restore particle lifetime if it was modified by user
269  fTrackManager->SetBackPDGLifetime(track);
270 
271  if (track->GetTrackStatus() != fSuspend) {
272 
273  // VMC application post track action
274  fMCApplication->PostTrack();
275 
276  // call post-tracking action of derived class
277  PostTrackingAction(track);
278  }
279 }
