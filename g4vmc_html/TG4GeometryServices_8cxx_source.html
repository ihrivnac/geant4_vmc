104 G4bool TG4GeometryServices::CompareElement(
105  G4double /*a*/, G4double z, const G4Element* element) const
106 {
109 
110  G4double ae = element->GetA() / TG4G3Units::AtomicWeight();
111  G4double ze = element->GetZ();
112 
113  // g3tog4 can redefine A

123  return true;

150 G4bool TG4GeometryServices::CompareMaterial(
151  G4int nofElements, G4double density, const G4Material* material) const
152 {
155 
156  G4double dm = material->GetDensity() / TG4G3Units::MassDensity();
157  G4int ne = material->GetNumberOfElements();
158 
159  // density percentual difference
160  G4double diff = std::abs(density - dm) / (density + dm) * 2.;
161 
162  if (nofElements == ne && diff < fgkDensityTolerance)
163  return true;
164  else
165  return false;
166 }

169 G4double* TG4GeometryServices::ConvertAtomWeight(
170  G4int nmat, G4double* a, G4double* wmat) const
171 {
175 
176  G4double* weight = new G4double[abs(nmat)];
177 
178  if (nmat < 0) {
179  G4double aMol = 0.;
180  G4int i;
181  for (i = 0; i < abs(nmat); i++) {
182  // total molecular weight
183  aMol += wmat[i] * a[i];
184  }
185  if (aMol == 0.) {
186  TG4Globals::Warning("TG4GeometryServices", "ConvertAtomWeight",
187  "Total molecular weight = 0.");
188  }
189  for (i = 0; i < abs(nmat); i++) {
190  // weight fractions
191  weight[i] = wmat[i] * a[i] / aMol;
192  }
193  }
194  else
195  for (G4int j = 0; j < nmat; j++) weight[j] = wmat[j];
196 
197  return weight;
198 }

429  const TG4StringVector& matNames, const TG4doubleVector& matWeights)

432

442  }
443 
444  // fill vector of materials
445  std::vector<G4Material*> matVector;
446  G4int im;
447  for (im = 0; im < nofMaterials; im++) {
448  // material

450  matVector.push_back(material);
451  }
452 
453  // create the mixed material
454  G4Material* mixture = new G4Material(name, density, nofMaterials);
455  for (im = 0; im < nofMaterials; im++) {
