116 G4bool TG4RegionsManager::Iterate(G4double energyCut, G4double& lowerCut,
117  G4double& higherCut, G4double defaultRangeCut, G4double lowEdgeEnergy,
118  G4double highEdgeEnergy, G4int nbin,
119  std::map<G4double, G4double>& energyToRangeMap,
120  std::map<G4double, G4double>::const_iterator& it, G4Material* material,
121  G4VRangeToEnergyConverter& converter) const
122 {
124 
125  G4double step = (higherCut - lowerCut) / nbin;
126  energyToRangeMap.clear();
127  G4String indent(" ");
128  for (G4int i = 0; i < nbin; i++) {
129  G4double rangeCut = lowerCut + i * step;
130  if (rangeCut < defaultRangeCut) continue;
131  G4double energy = converter.Convert(rangeCut, material);
132  if (VerboseLevel() > 2) {
133  G4cout << indent << "For range: " << rangeCut << " mm got energy "
134  << energy / MeV << " MeV" << G4endl;
135  }
136 
137  if (energy < lowEdgeEnergy) energy = lowEdgeEnergy;
138  if (energy > highEdgeEnergy) energy = highEdgeEnergy;
139 
140  energyToRangeMap.insert(
141  std::pair<G4double, G4double>(energy / MeV, rangeCut));
142  }
143 
144  // Now find the closest value of energy
145  it = energyToRangeMap.lower_bound(energyCut);
146 
147  if (it == energyToRangeMap.end()) {
148  if (VerboseLevel() > 2) {
149  G4cout << indent << "Range not found " << G4endl;
150  }
151  return false;
152  }
153  else {
154  if (VerboseLevel() > 2) {
155  G4cout << indent << "Found range limit: " << it->second << " mm"
156  << G4endl;
157  }
158  return true;
159  }
160 }

163 G4double TG4RegionsManager::ConvertEnergyToRange(G4double energyCut,
164  G4Material* material, G4VRangeToEnergyConverter& converter,
165  G4double defaultRangeCut) const
166 {
169 
170  G4double lowEdgeEnergy = converter.GetLowEdgeEnergy();
171  G4double highEdgeEnergy = converter.GetHighEdgeEnergy();
172 
173  // First compute default energy cut
174  std::map<G4double, G4double> energyToRangeMap;
175  G4double energy = converter.Convert(defaultRangeCut, material);
176  if (energy < lowEdgeEnergy) energy = lowEdgeEnergy;
177  if (energy > highEdgeEnergy) energy = highEdgeEnergy;
178  energyToRangeMap.insert(
179  std::pair<G4double, G4double>(energy / MeV, defaultRangeCut));
180 
181  // Compute the range order where we will start iterating
182  // withing the orders in between fgkMinRangeOrder and fgkMaxRangeOrder
183  // (1e-03mm, 1e-02, 1e-01, 1mm, 1cm, 10 cm, 1m, 10m, 100m, 1000m )
184  G4String indent(" ");
185  for (G4int i = fgkMinRangeOrder; i <= fgkMaxRangeOrder; i++) {
186  G4double rangeCut = pow(10., i);
187  if (rangeCut < defaultRangeCut) continue;
188  energy = converter.Convert(rangeCut, material);
189  if (VerboseLevel() > 2) {
190  G4cout << indent << "For range: " << std::setw(5) << rangeCut
191  << " mm got energy " << energy / MeV << " MeV" << G4endl;
192  }
193 
194  if (energy < lowEdgeEnergy) energy = lowEdgeEnergy;
195  if (energy > highEdgeEnergy) energy = highEdgeEnergy;
196  energyToRangeMap.insert(
197  std::pair<G4double, G4double>(energy / MeV, rangeCut));
198  }
199 
200  // If energyCut is above the maximum range order return the last
201  // computed value
202  if (energyToRangeMap.rbegin()->first < energyCut) {
203  if (VerboseLevel() > 2) {
204  G4cout << indent << "Outside range, return the highest value " << G4endl;
205  }
206  return energyToRangeMap.rbegin()->second;
207  }
208 
209  // Now find the closest value of energy which is
210  // equal or higher than given energyCut
211  std::map<G4double, G4double>::const_iterator it =
212  energyToRangeMap.lower_bound(energyCut);
213 
214  if (it == energyToRangeMap.end()) {
215  if (VerboseLevel() > 2) {
216  G4cout << indent << "Range not found " << G4endl;
217  }
218  return -1.0;
219  }
220 
221  if (VerboseLevel() > 2) {
222  G4cout << indent << "Found range limit: " << it->second << " mm" << G4endl;
223  }
224 
225  // Now iterate up to desired precision of range
226  G4int iteration = 0;
227  while (iteration < fRangePrecision) {
228  G4int nbin = fgkNofBins;
229  if (iteration == 0) --nbin;
230 
231  G4double higherCut = it->second;
232  // G4cout << "higherCut = " << higherCut << G4endl;
233  if (it == energyToRangeMap.begin()) return higherCut;
234 
235  --it;
236  G4double lowerCut = it->second;
237  // G4cout << "lowerCut = " << lowerCut << G4endl;
238  G4bool found =
239  Iterate(energyCut, lowerCut, higherCut, defaultRangeCut, lowEdgeEnergy,
240  highEdgeEnergy, nbin, energyToRangeMap, it, material, converter);
241 
242  if (!found) {
243  // Now we have to go one step back to get below the user cut value
244  // unless we are at the beginning of the map
245  if (it == energyToRangeMap.begin()) return higherCut;
246  --it;
247  return it->second;
248  ;
249  }
250  ++iteration;
251  }
252 
253  // Now we have to go one step back to get below the user cut value
254  // unless we are at the beginning of the map
255  if (it == energyToRangeMap.begin()) return it->second;
256  --it;
257  return it->second;
258  ;
259 }

326 G4bool TG4RegionsManager::CheckCut(TG4Limits* limits, TG4G3Cut cut,
327  const G4String& particleName, G4double range, G4double energy) const
328 {
331 
332  G4double energyLimits = GetEnergyCut(limits, cut, DBL_MAX);
333  if (energyLimits == DBL_MAX) return true;
334 
335  if (VerboseLevel() > 1) {
336  G4cout << " " << particleName << " cut:" << G4endl << " range = " << range
337  << " mm" << G4endl << " energy from range = " << energy / MeV
338  << " MeV"
339  << " energy from limits = " << energyLimits << " MeV" << G4endl;
340  }
341 
342  // Get default range cut value from physics list
343  G4double defaultRangeCut = 0.;
344  if (particleName == "e-") {
345  defaultRangeCut = TG4PhysicsManager::Instance()->GetCutForElectron();
346  }
347  else if (particleName == "gamma") {
348  defaultRangeCut = TG4PhysicsManager::Instance()->GetCutForGamma();
349  }
350 
351  if (energy > energyLimits * (1.0 + fgkEnergyTolerance) &&
352  fabs(range - defaultRangeCut) > fgkRangeTolerance) {
353  G4cout << " !! " << particleName << " cut from range > cut from limits !!"
354  << G4endl;
355  if (VerboseLevel() <= 1) {
356  G4cout << " energy from range = " << energy / MeV << " MeV"
357  << " energy from limits = " << energyLimits << " MeV" << G4endl;
358  }
359  return false;
360  }
361 
362  return true;
363 }

476 {
478 
479  if (VerboseLevel() > 0) G4cout << "Converting VMC cuts in regions" << G4endl;
480 
481  // Do nothing if no option is activated
482  if (!(fApplyForGamma || fApplyForElectron || fApplyForPositron ||
483  fApplyForProton)) {
484  return;
485  }
486 
487  // Create G4 range to energy converters
488  G4RToEConvForElectron g4ConverterEle;
489  G4RToEConvForGamma g4ConverterGam;
490 
491  // Get default range cut values from physics manager
492  G4double defaultRangeCutEle =
493  TG4PhysicsManager::Instance()->GetCutForElectron();
494  G4double defaultRangeCutGam = TG4PhysicsManager::Instance()->GetCutForGamma();
495  G4double defaultRangeCutPositron =
496  TG4PhysicsManager::Instance()->GetCutForPositron();
497  G4double defaultRangeCutProton =
498  TG4PhysicsManager::Instance()->GetCutForProton();
499 
500  // Create a new region with default cuts
501  G4Region* defaultRegion = new G4Region(fgkDefaultRegionName);
502  G4ProductionCuts* dcuts = new G4ProductionCuts();
503  dcuts->SetProductionCut(defaultRangeCutGam, 0);
504  dcuts->SetProductionCut(defaultRangeCutEle, 1);
505  dcuts->SetProductionCut(defaultRangeCutPositron, 2);
506  dcuts->SetProductionCut(defaultRangeCutProton, 3);
507  defaultRegion->SetProductionCuts(dcuts);
508 
509  // Get world default region
510  G4LogicalVolume* worldLV =
511  TG4GeometryServices::Instance()->GetWorld()->GetLogicalVolume();
512  G4Region* worldRegion = worldLV->GetRegion();
513 
514  // Get global energy cuts
515  G4double cutEleGlobal = GetGlobalEnergyCut(kCUTELE);
516  G4double cutGamGlobal = GetGlobalEnergyCut(kCUTGAM);
517  if (VerboseLevel() > 2) {
518  G4cout << "Global cut values: "
519  << " CUTELE = " << cutEleGlobal << " MeV"
520  << " CUTGAM = " << cutGamGlobal << " MeV" << G4endl;
521  }
522 
523  G4int counter = 0;
524  std::set<G4Material*> processedMaterials;
525 
526  // Define region for each logical volume
527  //
528 
529  G4LogicalVolumeStore* lvStore = G4LogicalVolumeStore::GetInstance();
530 
531  for (G4int i = 0; i < G4int(lvStore->size()); i++) {
532 
533  G4LogicalVolume* lv = (*lvStore)[i];
534  G4bool isWorld = (lv == worldLV);
535 
536  TG4Medium* medium =
537  TG4GeometryServices::Instance()->GetMediumMap()->GetMedium(lv, false);
538  if (!medium) continue;
539 
540  G4Material* material = lv->GetMaterial();
541  if (VerboseLevel() > 1) {
542  G4cout << "-- Volume = " << i << " " << lv << " " << lv->GetName()
543  << " material = " << material->GetName() << G4endl;
544  }
545 
546  // Get region, if it already exists, and add the logical volume
547  G4String regionName = material->GetName();
548  G4Region* region =
549  G4RegionStore::GetInstance()->GetRegion(regionName, false);
550  if (region && !isWorld) {
551  if (VerboseLevel() > 1) {
552  G4cout << " "
553  << "adding volume in region = " << regionName << G4endl;
554  }
555  if (lv->GetRegion() != region) region->AddRootLogicalVolume(lv);
556  }
557 
558  // If this material was already processed and did not result
559  // in a new region: add the logical volume to the default region
560  if (processedMaterials.find(material) != processedMaterials.end() &&
561  !isWorld) {
562  if (VerboseLevel() > 1) {
563  G4cout << " "
564  << "adding volume in the default region" << G4endl;
565  }
566  defaultRegion->AddRootLogicalVolume(lv);
567  continue;
568  }
569 
570  // Get energy cuts defined in limits
571  TG4Limits* limits = (TG4Limits*)lv->GetUserLimits();
572  G4double cutEle = GetEnergyCut(limits, kCUTELE, cutEleGlobal);
573  G4double cutGam = GetEnergyCut(limits, kCUTGAM, cutGamGlobal);
574 
575  // Convert energy cuts defined in limits in range cuts
576  G4double rangeEle =
577  GetRangeCut(cutEle, material, g4ConverterEle, defaultRangeCutEle);
578  G4double rangeGam =
579  GetRangeCut(cutGam, material, g4ConverterGam, defaultRangeCutGam);
580  if (VerboseLevel() > 1) {
581  G4cout << ".. converted in e- rangeCut = " << rangeEle << " mm "
582  << "gamma rangeCut = " << rangeGam << " mm" << G4endl;
583  }
584 
585  if (fabs(rangeEle - defaultRangeCutEle) < 1e-03 &&
586  fabs(rangeGam - defaultRangeCutGam) < 1e-03 && !region) {
587  // Do not create a new region if range cuts do not differ
588  // from those in the default region and region was not defined
589  // before
590  if (!isWorld) {
591  if (VerboseLevel() > 1) {
592  G4cout << " "
593  << "adding volume in the default region" << G4endl;
594  }
595  defaultRegion->AddRootLogicalVolume(lv);
596  }
597  processedMaterials.insert(material);
598  }
599  else {
600  // Create new production cuts
601  G4ProductionCuts* cuts = new G4ProductionCuts();
602 
603  if (fApplyForGamma) {
604  cuts->SetProductionCut(rangeGam, 0);
605  }
606  else {
607  cuts->SetProductionCut(defaultRangeCutGam, 0);
608  }
609 
610  if (fApplyForElectron) {
611  cuts->SetProductionCut(rangeEle, 1);
612  }
613  else {
614  cuts->SetProductionCut(defaultRangeCutEle, 1);
615  }
616 
617  if (fApplyForPositron) {
618  cuts->SetProductionCut(rangeEle, 2);
619  }
620  else {
621  cuts->SetProductionCut(defaultRangeCutPositron, 2);
622  }
623 
624  if (fApplyForProton) {
625  cuts->SetProductionCut(rangeEle, 3);
626  }
627  else {
628  cuts->SetProductionCut(defaultRangeCutProton, 2);
629  }
630 
631  if (isWorld) {
632  // set new production cuts to the world
633  worldRegion->SetProductionCuts(cuts);
634  worldRegion->RegionModified(true);
635  if (VerboseLevel() > 1) {
636  G4cout << " "
637  << "setting new production cuts to the world region" << G4endl;
638  }
639  }
640  else if (region) {
641  // set new production cuts to the existing region
642  region->SetProductionCuts(cuts);
643  region->RegionModified(true);
644  if (VerboseLevel() > 1) {
645  G4cout << " "
646  << "setting new production cuts to the existing region "
647  << regionName << G4endl;
648  }
649  }
650  else {
651  // create new region with new production cuts
652  region = new G4Region(regionName);
653  ++counter;
654  if (VerboseLevel() > 1) {
655  G4cout << " "
656  << "adding volume in a new region " << regionName << G4endl;
657  }
658  region->AddRootLogicalVolume(lv);
659  region->SetProductionCuts(cuts);
660  }
661  }
662  }
663 
664  if (VerboseLevel() > 0) {
665  G4cout << "Number of added regions: " << counter << G4endl;
666  }
667 }
