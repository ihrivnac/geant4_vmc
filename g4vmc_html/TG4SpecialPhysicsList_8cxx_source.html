140 //_____________________________________________________________________________
141 void TG4SpecialPhysicsList::Configure(const G4String& selection)
142 {
145 
146  Int_t tg4VerboseLevel = TG4Verbose::VerboseLevel();
147 
148  RegisterPhysics(new TG4TransitionRadiationPhysics(tg4VerboseLevel));
149  RegisterPhysics(new TG4ProcessControlMapPhysics(tg4VerboseLevel));
150 
151  G4int itoken = 0;
152  TString token = TG4Globals::GetToken(itoken, selection);
153  G4bool isGflash = false;
154  while (token != "") {
155 
156  if (token == "specialCuts") {
157  // G4cout << "Registering special cuts physics" << G4endl;
158  RegisterPhysics(new TG4SpecialCutsPhysics(tg4VerboseLevel));
159  fIsSpecialCuts = true;
160  }
161  else if (token == "stepLimiter") {
162  // G4cout << "Registering step limiter physics" << G4endl;
163  RegisterPhysics(new TG4StepLimiterPhysics(tg4VerboseLevel));
164  }
165  else if (token == "stackPopper") {
166  // G4cout << "Registering stack popper physics" << G4endl;
167  fStackPopperPhysics = new TG4StackPopperPhysics(tg4VerboseLevel);
168  RegisterPhysics(fStackPopperPhysics);
169  }
170  else if (token == "gflash") {
171  isGflash = true;
172  }
173  else {
174  TG4Globals::Warning(
175  "TG4SpecialPhysicsList", "Configure", "Unrecognized option " + token);
176  }
177  token = TG4Globals::GetToken(++itoken, selection);
178  }
179  RegisterPhysics(new TG4UserParticlesPhysics(tg4VerboseLevel));
180  RegisterPhysics(new TG4ExtDecayerPhysics(tg4VerboseLevel));
181  RegisterPhysics(new TG4ProcessMCMapPhysics(tg4VerboseLevel));
182 
183  fEmModelPhysics = new TG4EmModelPhysics(tg4VerboseLevel);
184  RegisterPhysics(fEmModelPhysics);
185  fFastSimulationPhysics = new TG4FastSimulationPhysics(tg4VerboseLevel);
186  RegisterPhysics(fFastSimulationPhysics);
187  if (isGflash) {
188  fFastSimulationPhysics->SetUserFastSimulation(
189  new TG4GflashFastSimulation());
190  }
191 }
