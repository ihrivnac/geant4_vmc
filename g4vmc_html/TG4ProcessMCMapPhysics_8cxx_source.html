258 void TG4ProcessMCMapPhysics::ConstructProcess()
259 {
262 
263  if (VerboseLevel() > 1) {
264  G4cout << "TG4ProcessMCMapPhysics::ConstructProcess: " << G4endl;
265  }
266 
267  TG4ProcessMCMap* mcMap = TG4ProcessMCMap::Instance();
268  G4bool success = true;
269 
270  auto aParticleIterator = GetParticleIterator();
271  aParticleIterator->reset();
272  while ((*aParticleIterator)()) {
273  // skip iteration if particle does not have a process manager
274  if (!aParticleIterator->value()->GetProcessManager()) continue;
275 
276  G4ProcessVector* processVector =
277  aParticleIterator->value()->GetProcessManager()->GetProcessList();
278 
279  for (G4int i = 0; i < processVector->length(); i++) {
280 
281  G4String processName = (*processVector)[i]->GetProcessName();
282 
283  if (mcMap->GetMCProcess(processName) == kPNoProcess
284 #if (!(ROOT_VERSION_CODE >= ROOT_VERSION(6, 7, 3)) && \
285  !((ROOT_VERSION_CODE <= ROOT_VERSION(6, 0, 0)) && \
286  (ROOT_VERSION_CODE >= ROOT_VERSION(5, 34, 34))))
287  && processName != "GammaXTRadiator" &&
288  processName != "StrawXTRadiator" && processName != "RegularXTRadiator"
289 #endif
290  ) {
291  G4String text = "Unknown process code for ";
292  text += processName;
293  TG4Globals::Warning("TG4ProcessMCMapPhysics", "ConstructProcess", text);
294  success = false;
295  }
296  }
297  }
298 
299  if (VerboseLevel() > 0) {
300  if (success)
301  G4cout << "### Processes mapped to VMC codes ok." << G4endl;
302  else
303  G4cout << "### Processes mapped to VMC codes: some proceesses unknown."
304  << G4endl;
305  }
306 }
