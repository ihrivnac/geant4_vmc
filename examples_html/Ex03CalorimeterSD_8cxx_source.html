132 void Ex03CalorimeterSD::Initialize()
133 {
134  /// Register hits collection in the Root manager;
135  /// set sensitive volumes.
136 
137  if (TMCRootManager::Instance()) Register();
138 
139  // Keep the pointer to TVirtualMC object as a data member
140  // to avoid a possible performance penalty due to a frequent retrieval
141  // from the thread-local storage
142  fMC = gMC;
143 
144  fAbsorberVolId = fMC->VolId("ABSO");
145  fGapVolId = fMC->VolId("GAPX");
146 
147  if (fAbsorberVolId == 0 && fGapVolId == 0) {
148  // Volume names are different in B4 detector construction
149  fAbsorberVolId = fMC->VolId("Abso");
150  fGapVolId = fMC->VolId("Gap");
151  }
152 }

154 //_____________________________________________________________________________
155 void Ex03CalorimeterSD::ProcessHits()
156 {
157  /// Account energy deposit and track lengths for each layer in its hit.
158 
159  Int_t copyNo;
160  Int_t id = fMC->CurrentVolID(copyNo);
161 
162  if (id != fAbsorberVolId && id != fGapVolId) return;
163 
164  fMC->CurrentVolOffID(2, copyNo);
165 
166  Double_t edep = fMC->Edep();
167 
168  Double_t step = 0.;
169  if (fMC->TrackCharge() != 0.) step = fMC->TrackStep();
170 
171  if (!GetHit(copyNo)) {
172  return;
173  }
174 
175  if (id == fAbsorberVolId) {
176  GetHit(copyNo)->AddAbs(edep, step);
177  }
178 
179  if (id == fGapVolId) {
180  GetHit(copyNo)->AddGap(edep, step);
181  }
182 }

219 //_____________________________________________________________________________
220 void Ex03CalorimeterSD::PrintTotal() const
221 {
222  /// Print the total values for all layers.
223 
224  Double_t totEAbs = 0.;
225  Double_t totLAbs = 0.;
226  Double_t totEGap = 0.;
227  Double_t totLGap = 0.;
228 
229  Int_t nofHits = fCalCollection->GetEntriesFast();
230  for (Int_t i = 0; i < nofHits; i++) {
231  totEAbs += GetHit(i)->GetEdepAbs();
232  totLAbs += GetHit(i)->GetTrakAbs();
233  totEGap += GetHit(i)->GetEdepGap();
234  totLGap += GetHit(i)->GetTrakGap();
235  }
236 
237  cout << " Absorber: total energy (MeV): " << setw(7) << totEAbs * 1.0e03
238  << " total track length (cm): " << setw(7) << totLAbs << endl
239  << " Gap: total energy (MeV): " << setw(7) << totEGap * 1.0e03
240  << " total track length (cm): " << setw(7) << totLGap << endl;
241 }
