302 void Ex06MCApplication::Stepping()
303 {
304  /// User actions at each step
305 
306  fVerbose.Stepping();
307 
308  // Work around for Fluka VMC, which does not call
309  // MCApplication::PreTrack()
310  //
311  static Int_t trackId = 0;
312  if (TString(gMC->GetName()) == "TFluka" &&
313  gMC->GetStack()->GetCurrentTrackNumber() != trackId) {
314 
315  fVerbose.PreTrack();
316  trackId = gMC->GetStack()->GetCurrentTrackNumber();
317  if (gMC->TrackPid() == 50000050) fGammaCounter++;
318  if (gMC->TrackPid() == 50000051) fFeedbackCounter++;
319  }
320 
321  if (!fTestStackPopper) return;
322 
323  // Stack popper test
324  // Add 1 feedback photon (50000051, fixed momentum) when a charged track
325  // enters TANK and 3 feedback photons (momentum in opposite direction to
326  // parent photon)
327  // when a photon is stopped in TANK
328 
329  // Charged particles entering in TANK
330  TString volName = gMC->CurrentVolName();
331  if (volName != "TANK") return;
332 
333  if ((gMC->TrackCharge() != 0.) && (gMC->IsTrackEntering())) {
334  // 1 keV
335  Double_t energy = 1e-06;
336  TLorentzVector momentum(energy, 0., 0., energy);
337  GenerateFeedback(1, momentum);
338  }
339 
340  if ((gMC->TrackPid() == 50000050) && (gMC->Edep() > 0.)) {
341  TLorentzVector momentum;
342  gMC->TrackMomentum(momentum);
343  momentum = -1. * momentum;
344  GenerateFeedback(3, momentum);
345  }
346 }

348 //_____________________________________________________________________________
349 void Ex06MCApplication::GenerateFeedback(
350  Int_t nofPhotons, TLorentzVector momentum)
351 {
352  /// Generate FeedBack photons for the current particle.
353 
354  for (Int_t i = 0; i < nofPhotons; ++i) {
355  // same position as the parent track
356  TLorentzVector position;
357  gMC->TrackPosition(position);
358  // Feedback photon
359  Int_t pdgEncoding = 50000051;
360  // Fixed polarization
361  Double_t polarization[3];
362  polarization[0] = 0.3;
363  polarization[1] = 0.4;
364  polarization[2] = 0.866025403784438597;
365 
366  Int_t ntrack;
367  gMC->GetStack()->PushTrack(1, gMC->GetStack()->GetCurrentTrackNumber(),
368  pdgEncoding, momentum.X(), momentum.Y(), momentum.Z(), momentum.T(),
369  position.X(), position.Y(), position.Z(), position.T(), polarization[0],
370  polarization[1], polarization[2], kPFeedBackPhoton, ntrack, 1.0, 0);
371  }
372 }
