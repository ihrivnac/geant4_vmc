90 G4double TG4RootNavigator::ComputeStep(const G4ThreeVector& pGlobalPoint,
91  const G4ThreeVector& pDirection, const G4double pCurrentProposedStepLength,
92  G4double& pNewSafety)
93 {
106 
107  // The following 2 lines are not needed if G4 calls first LocateGlobalPoint...
108  // fGeometry->ResetState();
109  static Long64_t istep = 0;
110  istep++;
111 
112 #ifdef G4ROOT_DEBUG
113  G4cout.precision(8);
114  G4cout << "*** ComputeStep #" << istep << ": ***"
115  << fHistory.GetTopVolume()->GetName()
116  << " entered: " << fEnteredDaughter << " exited: " << fExitedMother
117  << G4endl;
118 #endif
119  Double_t tol = 0.;
120  Bool_t compute_safety = kTRUE;
121  Double_t pstep = pCurrentProposedStepLength;
122  if (pstep > TGeoShape::Big()) {
123  pstep = TGeoShape::Big();
124  compute_safety = kFALSE;
125  pNewSafety = 0.0;
126  }
127 
128  Bool_t frombdr = fEnteredDaughter | fExitedMother;
129 #ifdef G4ROOT_DEBUG
130  Bool_t oldpoint = kFALSE;
131 #endif
132  if (frombdr) {
133  Double_t npt[3];
134  tol = TGeoShape::Tolerance();
135  npt[0] = pGlobalPoint.x() * gCm + tol * pDirection.x();
136  npt[1] = pGlobalPoint.y() * gCm + tol * pDirection.y();
137  npt[2] = pGlobalPoint.z() * gCm + tol * pDirection.z();
138  fNavigator->SetCurrentPoint(npt[0], npt[1], npt[2]);
139  compute_safety = kFALSE;
140  pNewSafety = 0.0;
141  }
142  else {
143  fNavigator->SetCurrentPoint(
144  pGlobalPoint.x() * gCm, pGlobalPoint.y() * gCm, pGlobalPoint.z() * gCm);
145  Double_t d2 = pGlobalPoint.diff2(fSafetyOrig);
146  if (d2 < 1.e-10) {
147 #ifdef G4ROOT_DEBUG
148  oldpoint = kTRUE;
149 #endif
150  compute_safety = kFALSE;
151  pNewSafety = fLastSafety;
152  }
153  fSafetyOrig = pGlobalPoint;
154  }
155  fNavigator->SetCurrentDirection(
156  pDirection.x(), pDirection.y(), pDirection.z());
157  fNavigator->FindNextBoundary(-(pstep * gCm - tol), "", !compute_safety);
158 
159  if (compute_safety) {
160  pNewSafety = (fNavigator->GetSafeDistance() - tol) * cm;
161  if (pNewSafety < 0.) pNewSafety = 0.;
162  fLastSafety = pNewSafety;
163  }
164  G4double step = (fNavigator->GetStep() + tol) * cm;
165  // if (step >= pCurrentProposedStepLength) step = kInfinity;
166  if (step < 1.e3 * tol * cm) {
167  step = 0.;
168  fNzeroSteps++;
169  // Geant4 will abandon the track if the number of zero steps>50 just
170  // because it expects a non-zero distance inside the mother to the next
171  // daughter The way out is to generate an extra very small fake step in the
172  // mother, before this threshold is reached
173  if (fNzeroSteps > gAbandonZeroSteps) step = gZeroStepThr;
174  }
175  else {
176  fNzeroSteps = 0;
177  }
178  fStepEntering = fNavigator->IsStepEntering();
179  fStepExiting = fNavigator->IsStepExiting();
180  if (fStepEntering || fStepExiting) {
181  fNextPoint = pGlobalPoint + step * pDirection;
182  }
183  else {
184  step = kInfinity;
185  }
186 
187 #ifdef G4ROOT_DEBUG
188  G4cout.precision(12);
189  G4cout << "ComputeStep: point=" << pGlobalPoint << " dir=" << pDirection
190  << G4endl;
191  G4cout << " pstep=" << pCurrentProposedStepLength
192  << " snext=" << step << G4endl;
193  G4cout << " safe =" << pNewSafety << " frombdr=" << frombdr
194  << " oldpoint=" << oldpoint << " entering=" << fStepEntering
195  << " exiting=" << fStepExiting << G4endl;
196  if (fStepEntering || fStepExiting && fNavigator->GetNextNode()) {
197  G4cout << " current: "
198  << fNavigator->GetCurrentNode()->GetName()
199  << " next: " << fNavigator->GetNextNode()->GetName() << G4endl;
200  }
201 #endif
202  return step;
203 }

241 TGeoNode* TG4RootNavigator::SynchronizeGeoManager()
242 {
246  Int_t geolevel = fNavigator->GetLevel();
247  Int_t depth = fHistory.GetDepth();
248  Int_t nodeIndex, level;
249  G4VPhysicalVolume* pvol;
250  TGeoNode *pnode, *newnode = 0;
251  for (level = 1; level <= depth; level++) {
252  pvol = fHistory.GetVolume(level);
253  newnode = fDetConstruction->GetNode(pvol);
254  if (level <= geolevel) {
255  // TGeo has also something at this level - check if it matches what is
256  // in fHistory
257  pnode = fNavigator->GetMother(geolevel - level);
258  // If the node at this level matches the one in the history, do nothing
259  if (pnode == newnode) continue;
260  // From this level down we need to update TGeo path.
261  while (geolevel >= level) {
262  fNavigator->CdUp();
263  geolevel--;
264  }
265  // Now TGeo is at level-1 and needs to update level
266  // this should be the index of the node to be used in CdDown(index)
267  nodeIndex = fNavigator->GetCurrentVolume()->GetIndex(newnode);
268  if (nodeIndex < 0) {
269  G4cerr << "SynchronizeGeoManager did not work (1)!!!" << G4endl;
270  return NULL;
271  }
272  // nodeIndex = fHistory.GetReplicaNo(level);
273  fNavigator->CdDown(nodeIndex);
274  geolevel++; // Should be equal to i now
275  }
276  else {
277  // This level has to be synchronized
278  nodeIndex = fNavigator->GetCurrentVolume()->GetIndex(newnode);
279  if (nodeIndex < 0) {
280  G4cerr << "SynchronizeGeoManager did not work (2)!!!" << G4endl;
281  return NULL;
282  }
283  // nodeIndex = fHistory.GetReplicaNo(level);
284  fNavigator->CdDown(nodeIndex);
285  geolevel++; // Should be equal to i now
286  }
287  }
288  return fNavigator->GetCurrentNode();
289 }

292 G4VPhysicalVolume* TG4RootNavigator::SynchronizeHistory()
293 {
297  Int_t depth = fHistory.GetDepth();
298  Int_t geolevel = fNavigator->GetLevel();
299  G4VPhysicalVolume *pvol, *pnewvol = 0;
300  TGeoNode* pnode;
301  Int_t level;
302  for (level = 0; level <= geolevel; level++) {
303  pnode = fNavigator->GetMother(geolevel - level);
304  pnewvol = fDetConstruction->GetG4VPhysicalVolume(pnode);
305  if (level <= depth) {
306  pvol = fHistory.GetVolume(level);
307  // If the phys. volume at this level matches the one in the history, do
308  // nothing
309  if (pvol == pnewvol) continue;
310  // From this level down we need to update G4 history.
311  if (level) {
312  fHistory.BackLevel(depth - level + 1);
313  // Now fHistory is at the level i-1 and needs to update level i
314  fHistory.NewLevel(pnewvol, kNormal, pnewvol->GetCopyNo());
315  }
316  else {
317  // We need to refresh top level
318  fHistory.BackLevel(depth);
319  fHistory.SetFirstEntry(pnewvol);
320  }
321  depth = level;
322  }
323  else {
324  // This level has to be added to the current history.
325  fHistory.NewLevel(pnewvol, kNormal, pnewvol->GetCopyNo());
326  depth++; // depth=level
327  }
328  }
329  if (depth > level - 1) fHistory.BackLevel(depth - level + 1);
330  if (fNavigator->IsOutside()) pnewvol = NULL;
331  return pnewvol;
332 }
