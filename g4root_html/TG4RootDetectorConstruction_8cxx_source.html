247 G4LogicalVolume* TG4RootDetectorConstruction::CreateG4LogicalVolume(
248  TGeoVolume* vol)
249 {
252  if (!vol) return NULL;
253  G4LogicalVolume* pVolume = GetG4Volume(vol);
254  if (pVolume) return pVolume;
255  G4String sname(vol->GetName());
256  G4VSolid* pSolid = CreateG4Solid(vol->GetShape());
257  if (!pSolid) {
258  G4ExceptionDescription description;
259  description << " "
260  << "Cannot make solid from shape: "
261  << vol->GetShape()->GetName();
262  G4Exception("TG4RootDetectorConstruction::CreateG4LogicalVolume",
263  "G4Root_F002", FatalException, description);
264  }
265  G4Material* pMaterial = 0;
266  if (vol->IsAssembly()) {
267  pMaterial =
268  GetG4Material((TGeoMaterial*)fGeometry->GetListOfMaterials()->At(0));
269  }
270  else {
271  pMaterial = GetG4Material(vol->GetMedium()->GetMaterial());
272  }
273  if (!pMaterial) {
274  G4ExceptionDescription description;
275  description << " "
276  << "Cannot make material for volume: " << vol->GetName()
277  << G4endl;
278  G4Exception("TG4RootDetectorConstruction::CreateG4LogicalVolume",
279  "G4Root_F003", FatalException, description);
280  }
281  pVolume =
282  new G4LogicalVolume(pSolid, pMaterial, sname, NULL, NULL, NULL, false);
283  fG4VolumeMap.insert(G4VolumeVal_t(vol, pVolume));
284  fVolumeMap.insert(VolumeVal_t(pVolume, vol));
285  return pVolume;
286 }

289 G4VPhysicalVolume* TG4RootDetectorConstruction::CreateG4PhysicalVolume(
290  TGeoNode* node)
291 {
293  if (!node) return NULL;
294  node->cd();
295  G4VPhysicalVolume* pPhysicalVolume = GetG4VPhysicalVolume(node);
296  if (pPhysicalVolume) return pPhysicalVolume;
297  TGeoMatrix* mat = node->GetMatrix();
298  const Double_t* tr = mat->GetTranslation();
299  G4ThreeVector tlate(tr[0] * cm, tr[1] * cm, tr[2] * cm);
300  G4RotationMatrix* pRot = CreateG4Rotation(mat);
301  G4String pName(node->GetVolume()->GetName());
302  G4LogicalVolume* pCurrentLogical = CreateG4LogicalVolume(node->GetVolume());
303  if (!pCurrentLogical) {
304  G4ExceptionDescription description;
305  description << " "
306  << "No G4 volume created for TGeo node " << node->GetName()
307  << G4endl;
308  G4Exception("TG4RootDetectorConstruction::CreateG4PhysicalVolume",
309  "G4Root_F004", FatalException, description);
310  }
311  G4LogicalVolume* pMotherLogical =
312  CreateG4LogicalVolume(node->GetMotherVolume());
313  if (!pMotherLogical && node != fGeometry->GetTopNode()) {
314  G4ExceptionDescription description;
315  description << " "
316  << "No G4 mother volume crated for TGeo node "
317  << node->GetName();
318  G4Exception("TG4RootDetectorConstruction::CreateG4PhysicalVolume",
319  "G4Root_F005", FatalException, description);
320  }
321  G4bool pMany = false;
322  G4int pCopyNo = node->GetNumber();
323 
324  pPhysicalVolume = new G4PVPlacement(
325  pRot, tlate, pCurrentLogical, pName, pMotherLogical, pMany, pCopyNo);
326  fG4PVolumeMap.insert(G4PVolumeVal_t(node, pPhysicalVolume));
327  fPVolumeMap.insert(PVolumeVal_t(pPhysicalVolume, node));
328  return pPhysicalVolume;
329 }

332 G4Material* TG4RootDetectorConstruction::CreateG4Material(
333  const TGeoMaterial* mat)
334 {
337  G4Material* pMaterial = GetG4Material(mat);
338  if (pMaterial) return pMaterial;
339  G4State state = kStateUndefined;
340  G4double temp = mat->GetTemperature();
341  G4double pressure = mat->GetPressure();
342  switch (mat->GetState()) {
343  case TGeoMaterial::kMatStateUndefined:
344  state = kStateUndefined;
345  break;
346  case TGeoMaterial::kMatStateSolid:
347  state = kStateSolid;
348  break;
349  case TGeoMaterial::kMatStateLiquid:
350  state = kStateLiquid;
351  break;
352  case TGeoMaterial::kMatStateGas:
353  state = kStateGas;
354  break;
355  }
356  G4String elname, symbol;
357  TGeoElementTable* table = fGeometry->GetElementTable();
358  G4String name(mat->GetName());
359  G4double density = mat->GetDensity() * (g / cm3);
360  if (density < universe_mean_density || mat->GetZ() < 1.) {
361  density = universe_mean_density;
362  pMaterial = new G4Material(name, 1., 1.01 * g / mole, density, kStateGas,
363  STP_Temperature, 3.e-18 * pascal);
364  fG4MaterialMap.insert(G4MaterialVal_t(mat, pMaterial));
365  // G4cout << pMaterial << G4endl;
366  return pMaterial;
367  }
368 
369  if (mat->IsMixture()) {
370  // Mixtures
371  const TGeoMixture* mixt = (const TGeoMixture*)mat;
372  G4int nComponents = mixt->GetNelements();
373  // G4cout << "Creating G4 mixture "<< name << G4endl;
374  pMaterial =
375  new G4Material(name, density, nComponents, state, temp, pressure);
376  for (Int_t i = 0; i < nComponents; i++) {
377  // TGeoElement *elem = mixt->GetElement(i);
378  // name = elem->GetTitle();
379  // G4Element *pElement = G4Element::GetElement(name);
380  TGeoElement* elem = table->GetElement(Int_t(mixt->GetZmixt()[i]));
381  if (!elem) {
382  G4ExceptionDescription description;
383  description << " "
384  << "Woops: no element corresponding to Z="
385  << Int_t(mixt->GetZmixt()[i]);
386  G4Exception("TG4RootDetectorConstruction::CreateG4Material",
387  "G4Root_F006", FatalException, description);
388  }
389  elname = elem->GetTitle();
390  symbol = elem->GetName();
391  G4Element* pElement =
392  new G4Element(elname, symbol, G4double(mixt->GetZmixt()[i]),
393  G4double(mixt->GetAmixt()[i]) * (g / mole));
394  pMaterial->AddElement(pElement, mixt->GetWmixt()[i]);
395  }
396  }
397  else {
398  // Materials with 1 element.
399  // G4cout << "Creating G4 material "<< name << G4endl;
400  pMaterial = new G4Material(name, G4double(mat->GetZ()),
401  mat->GetA() * g / mole, density, state, temp, pressure);
402  }
403  fG4MaterialMap.insert(G4MaterialVal_t(mat, pMaterial));
404  // G4cout << pMaterial << G4endl;
405  return pMaterial;
406 }
